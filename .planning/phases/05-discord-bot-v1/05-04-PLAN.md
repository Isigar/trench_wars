The outbound endpoint is the trickiest piece — Pattern 4's atomic claim semantics ensure two bot replicas (multi-region or rolling-deploy) cannot pick up the same outbound row twice. The lockForUpdate + UPDATE attempts++ in a single transaction is non-negotiable.

BotApiMatchSignupController signature:

  namespace App\Http\Controllers\BotApi
  use App\Models\GameMatch        // D-04-03-A direct import; NO alias
  use App\Models\GameRole
  use App\Services\MatchSignupService
  use App\Http\Requests\StoreBotMatchSignupRequest
  final class BotApiMatchSignupController extends Controller

  public function store(StoreBotMatchSignupRequest $request, GameMatch $match): JsonResponse {
      $gameRole = GameRole::findOrFail($request->validated('game_role_id'));
      $user = auth()->user(); // rebound by bot.acts-as middleware to the human Discord user
      try {
          $slot = app(MatchSignupService::class)->signup($match, $user, $gameRole);
          return response()->json(['slot' => MatchSlotData::from($slot)], 201);
      } catch (MatchNotOpenException $e) {
          return response()->json(['error'=>'match_not_open','message'=>__('bot.errors.match_not_open')], 422);
      } catch (TagRestrictedException $e) {
          return response()->json(['error'=>'tag_restricted','message'=>__('bot.errors.tag_restricted')], 422);
      } catch (AlreadySignedUpException $e) {
          return response()->json(['error'=>'already_signed_up','message'=>__('bot.errors.already_signed_up')], 422);
      } catch (CapacityExceededException $e) {
          return response()->json(['error'=>'capacity_full','message'=>__('bot.errors.capacity_full')], 422);
      }
  }

  public function destroy(GameMatch $match, GameRole $gameRole): JsonResponse {
      $user = auth()->user();
      // D-010-aligned destroy: lockForUpdate inside DB::transaction, clear occupant_user_id + confirmed_at
      $deleted = DB::transaction(function () use ($match, $gameRole, $user) {
          $slot = MatchSlot::where('match_id', $match->id)->where('game_role_id', $gameRole->id)
              ->where('occupant_user_id', $user->id)->lockForUpdate()->first();
          if ($slot === null) return 0;
          $slot->update(['occupant_user_id' => null, 'confirmed_at' => null]);
          return 1;
      });
      if ($deleted === 0) return response()->json(['error'=>'not_signed_up'], 404);
      return response()->json([], 204);
  }

BotApiOutboundController — RESEARCH Pattern 4 verbatim atomic claim:

  use App\Models\DiscordOutboundMessage
  use Illuminate\Support\Facades\DB

  public function pending(Request $request): JsonResponse {
      $limit = max(1, min(50, (int) $request->query('limit', 20)));
      $rows = DB::transaction(function () use ($limit) {
          $pending = DiscordOutboundMessage::dispatchable()
              ->orderBy('created_at')
              ->lockForUpdate()
              ->limit($limit)
              ->get();
          foreach ($pending as $row) {
              $row->update(['status' => 'dispatching', 'attempts' => $row->attempts + 1]);
          }
          return $pending->fresh();
      });
      return response()->json($rows);
  }

  public function markSent(MarkOutboundSentRequest $request, string $id): JsonResponse {
      $row = DiscordOutboundMessage::findOrFail($id);
      if ($row->status !== 'dispatching') {
          return response()->json(['error'=>'outbound_not_dispatching','message'=>__('bot.errors.outbound_not_dispatching')], 422);
      }
      $row->update(['status' => 'sent', 'sent_message_id' => $request->validated('sent_message_id'), 'last_error' => null]);
      return response()->json($row->fresh());
  }

  public function markFailed(MarkOutboundFailedRequest $request, string $id): JsonResponse {
      $row = DiscordOutboundMessage::findOrFail($id);
      if ($row->status !== 'dispatching') {
          return response()->json(['error'=>'outbound_not_dispatching','message'=>__('bot.errors.outbound_not_dispatching')], 422);
      }
      $backoffSeconds = [1, 5, 15, 60, 300][$row->attempts - 1] ?? 600;
      if ($row->attempts >= 3) {
          $row->update(['status' => 'failed', 'last_error' => $request->validated('last_error')]);
      } else {
          $row->update(['status' => 'pending', 'last_error' => $request->validated('last_error'), 'backoff_until' => now()->addSeconds($backoffSeconds)]);
      }
      return response()->json($row->fresh());
  }

BotApiDiscordEventController — Pitfall 10 echo suppression:

  use App\Models\Clan
  use App\Models\ClanMembership
  use App\Models\User
  use App\Models\DiscordOutboundMessage

  public function roleChange(RoleChangeEventRequest $request): JsonResponse {
      $data = $request->validated();
      // Pitfall 10: 60s echo window
      $echo = DiscordOutboundMessage::query()
          ->where('message_type', 'role_sync')
          ->where('status', 'sent')
          ->where('updated_at', '>', now()->subSeconds(60))
          ->where('payload->discord_user_id', $data['user_discord_id'])
          ->where('payload->discord_role_id', $data['role_discord_id'])
          ->where('payload->action', $data['action'])
          ->exists();
      if ($echo) {
          return response()->json(['action'=>'noop','reason'=>'own_echo','message'=>__('bot.errors.echo_suppressed')]);
      }
      // Idempotent reconcile
      $user = User::where('discord_id', $data['user_discord_id'])->first();
      $clan = Clan::where('discord_role_id', $data['role_discord_id'])->first();
      if ($user === null || $clan === null) {
          return response()->json(['action'=>'noop','reason'=>'unmapped']);
      }
      if ($data['action'] === 'add') {
          // Create ClanMembership if not active — defer to existing flow OR insert directly with a system causer (v1: insert)
          ClanMembership::firstOrCreate(
              ['user_id'=>$user->id,'clan_id'=>$clan->id,'left_at'=>null],
              ['role'=>'member','joined_at'=>now()]
          );
          return response()->json(['action'=>'created','clan_id'=>$clan->id]);
      }
      // remove
      $membership = ClanMembership::where('user_id',$user->id)->where('clan_id',$clan->id)->whereNull('left_at')->first();
      if ($membership) $membership->update(['left_at'=>now()]);
      return response()->json(['action'=>'ended']);
  }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Bot client -> /api/bot/* routes | Three-layer gate: Sanctum bearer + abilities + (where applicable) bot.acts-as |
| FormRequest -> Controller | All POST/DELETE bodies validated at route entry; controllers never read $request->input() directly |
