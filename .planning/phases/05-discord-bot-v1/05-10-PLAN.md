  import { EmbedBuilder, ActionRowBuilder, ButtonBuilder, ButtonStyle } from 'discord.js';
  import type { PublicMatchData, ClanData, PublicPlayerData } from '../types/apiContracts';
  import { encodeButtonId } from './customIds';
  import { statusColor } from './colors';

  export function matchCard(m: PublicMatchData): {
      embeds: EmbedBuilder[];
      components: ActionRowBuilder<ButtonBuilder>[];
  } {
      const title = m.title?.en ?? `${m.host_clan?.name ?? 'TBD'} match`;
      const scheduledTs = m.scheduled_at !== null ? Math.floor(new Date(m.scheduled_at).getTime() / 1000) : null;

      const embed = new EmbedBuilder()
          .setColor(statusColor(m.status))
          .setTitle(title)
          .setDescription(`**Type:** ${m.game_match_type?.display_name?.en ?? '—'}\n**Status:** ${m.status}`)
          .addFields(
              ...(scheduledTs !== null ? [{ name: 'Scheduled', value: `<t:${scheduledTs}:F>`, inline: true }] : []),
              { name: 'Host', value: m.host_clan?.name ?? '—', inline: true },
              { name: 'Slot summary', value: renderSlotSummary(m), inline: false },
          )
          .setFooter({ text: `Match id: ${m.id}` })
          .setTimestamp();

      const components: ActionRowBuilder<ButtonBuilder>[] = [];
      if (m.status === 'open') {
          const signupBtn = new ButtonBuilder()
              .setCustomId(encodeButtonId({ kind: 'match_open_signup_modal', matchId: m.id }))
              .setLabel('Sign up')
              .setStyle(ButtonStyle.Success);
          // 'Leave' button requires the role uuid which we don't have at the embed level;
          // user invokes /match leave with explicit role UUID, or signupModal handler offers a follow-up.
          components.push(new ActionRowBuilder<ButtonBuilder>().addComponents(signupBtn));
      }
      return { embeds: [embed], components };
  }

  function renderSlotSummary(m: PublicMatchData): string {
      if (!m.slots || m.slots.length === 0) return '*(no slots materialised yet)*';
      // Group by game_role_id and emit "Role: filled/total"
      const byRole = new Map<string, { name: string; total: number; filled: number }>();
      for (const slot of m.slots) {
          const key = slot.game_role_id;
          const entry = byRole.get(key) ?? { name: slot.role?.display_name?.en ?? key, total: 0, filled: 0 };
          entry.total += 1;
          if (slot.occupant_user_id !== null) entry.filled += 1;
          byRole.set(key, entry);
      }
      return Array.from(byRole.values())
          .map((r) => `**${r.name}**: ${r.filled}/${r.total}`)
          .join('\n')
          .slice(0, 1024); // Discord field value max 1024 chars
  }

  export function clanCard(c: ClanData): { embeds: EmbedBuilder[] } {
      const embed = new EmbedBuilder()
          .setColor(0x0078d4) // brand blue
          .setTitle(c.name ?? c.slug)
          .setDescription((c.description?.en ?? '').slice(0, 2000))
          .addFields(
              { name: 'Slug', value: c.slug, inline: true },
              ...(c.tags && c.tags.length > 0 ? [{ name: 'Tags', value: c.tags.map((t) => `\`${t.slug}\``).join(' '), inline: true }] : []),
          )
          .setFooter({ text: `Clan id: ${c.id}` });
      return { embeds: [embed] };
  }

  export function profileCard(p: PublicPlayerData): { embeds: EmbedBuilder[] } {
      const embed = new EmbedBuilder()
          .setColor(0x666666)
          .setTitle(`Player: ${p.slug}`);
      const fields: Array<{ name: string; value: string; inline?: boolean }> = [];
      // Privacy gate already collapsed private fields to null on the web side
      if (p.real_name !== null && p.real_name !== undefined) fields.push({ name: 'Real name', value: p.real_name, inline: true });
      if (p.discord_tag !== null && p.discord_tag !== undefined) fields.push({ name: 'Discord', value: p.discord_tag, inline: true });
      if (fields.length > 0) embed.addFields(...fields);
      embed.setFooter({ text: `Player id: ${p.id}` });
      return { embeds: [embed] };
  }

  import { ButtonInteraction, ModalBuilder, TextInputBuilder, TextInputStyle, ActionRowBuilder, MessageFlags } from 'discord.js';
  import { decodeButtonId, type ButtonAction } from '../lib/customIds';
  import { api } from '../services/api';

  export async function handle(interaction: ButtonInteraction): Promise<void> {
      const decoded = decodeButtonId(interaction.customId);
      if (decoded === null) {
          await interaction.reply({ content: 'Unknown button.', flags: MessageFlags.Ephemeral });
          return;
      }

      if (decoded.kind === 'match_open_signup_modal') {
          // Pop the modal — must be the initial response (Pitfall 1 corollary)
          const modal = new ModalBuilder()
              .setCustomId(`m:o:${decoded.matchId}`)
              .setTitle('Sign up to match')
              .addComponents(
                  new ActionRowBuilder<TextInputBuilder>().addComponents(
                      new TextInputBuilder()
                          .setCustomId('role')
                          .setLabel('Role UUID (from /match info)')
                          .setStyle(TextInputStyle.Short)
                          .setRequired(true),
                  ),
              );
          await interaction.showModal(modal);
          return;
      }

      // signup + leave have already-deferred interactions per interactionCreate dispatcher (Pitfall 11)
      // BUT showModal-bound interactions cannot have been pre-deferred at the dispatcher level —
      // refactor in interactionCreate.ts to defer ONLY non-modal-opening buttons. See task 2.

      if (decoded.kind === 'match_signup') {
          try {
              await api.post(`/matches/${decoded.matchId}/signups`,
                  { game_role_id: decoded.gameRoleId },
                  { actsAsDiscordId: interaction.user.id },
              );
              await interaction.editReply('Signed up.');
          } catch (err) {
              await interaction.editReply(translateError(err));
          }
          return;
      }

      if (decoded.kind === 'match_leave') {
          try {
              await api.delete(`/matches/${decoded.matchId}/signups/${decoded.gameRoleId}`,
                  { actsAsDiscordId: interaction.user.id },
              );
              await interaction.editReply('Left the match.');
          } catch (err) {
              await interaction.editReply(translateError(err));
          }
          return;
      }

      // clan_apply variant — v1 redirect
      if (decoded.kind === 'clan_apply') {
          await interaction.editReply('Apply via the website.');
          return;
      }
  }

  function translateError(err: unknown): string {
      const msg = (err as Error).message ?? 'Unknown error.';
      // The web API returns 4 typed error keys; the Error message from api.ts contains them
      if (msg.includes('match_not_open')) return 'This match is not open for signups.';
      if (msg.includes('capacity_full')) return 'This role is full.';
      if (msg.includes('tag_restricted')) return 'Your clan tags are not permitted on this match.';
      if (msg.includes('already_signed_up')) return 'You are already signed up to this match.';
      return msg.slice(0, 200);
  }

apps/bot/src/components/signupModal.ts:

  import { ModalSubmitInteraction } from 'discord.js';
  import { api } from '../services/api';

  // customId shape: 'm:o:<match_id>' — matches the encodeButtonId('match_open_signup_modal') format
  export async function handle(interaction: ModalSubmitInteraction): Promise<void> {
      const parts = interaction.customId.split(':');
      if (parts[0] !== 'm' || parts[1] !== 'o' || parts.length !== 3) {
          await interaction.editReply('Unknown modal.');
          return;
      }
      const matchId = parts[2]!;
      const roleId = interaction.fields.getTextInputValue('role').trim();

      // Validate role uuid shape before hitting the API (defensive)
      if (!/^[0-9a-f-]{36}$/i.test(roleId)) {
          await interaction.editReply('Invalid role UUID.');
          return;
      }

      try {
          await api.post(`/matches/${matchId}/signups`,
              { game_role_id: roleId },
              { actsAsDiscordId: interaction.user.id },
          );
          await interaction.editReply('Signed up to the match.');
      } catch (err) {
          await interaction.editReply(translateError(err));
      }
  }

  function translateError(err: unknown): string {
      const msg = (err as Error).message ?? 'Unknown error.';
      if (msg.includes('match_not_open')) return 'This match is not open for signups.';
      if (msg.includes('capacity_full')) return 'This role is full.';
      if (msg.includes('tag_restricted')) return 'Your clan tags are not permitted on this match.';
      if (msg.includes('already_signed_up')) return 'You are already signed up to this match.';
      return msg.slice(0, 200);
  }
