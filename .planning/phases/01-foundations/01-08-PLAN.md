**8. Update `apps/web/app/Http/Middleware/HandleInertiaRequests.php`** to share `locale` + `translations`:

```php
<?php

declare(strict_types=1);

namespace App\Http\Middleware;

use Illuminate\Http\Request;
use Inertia\Middleware;
use Tighten\Ziggy\Ziggy;

class HandleInertiaRequests extends Middleware
{
    protected $rootView = 'app';

    public function version(Request $request): ?string
    {
        return parent::version($request);
    }

    /**
     * @return array<string, mixed>
     */
    public function share(Request $request): array
    {
        return array_merge(parent::share($request), [
            'auth' => fn () => $request->user()?->only(['id', 'discord_id', 'username', 'avatar_url']),

            'locale'       => fn () => app()->getLocale(),
            'translations' => fn () => $this->translations(app()->getLocale()),

            'flash' => [
                'success' => fn () => $request->session()->get('success'),
                'error'   => fn () => $request->session()->get('error'),
            ],

            'ziggy' => fn (): array => [
                ...(new Ziggy)->toArray(),
                'location' => $request->url(),
            ],
        ]);
    }

    /**
     * Build the flat namespace→key dictionary from PHP lang files.
     *
     * Output shape: { 'auth.discord.button_label': 'Log in with Discord', ... }
     *
     * @return array<string, string>
     */
    protected function translations(string $locale): array
    {
        $namespaces = (array) config('i18n.shared_namespaces', ['auth', 'common', 'admin', 'home', 'validation']);

        $flat = [];
        foreach ($namespaces as $ns) {
            $values = (array) trans($ns, [], $locale);
            $this->flatten($flat, $ns, $values);
        }

        return $flat;
    }

    /** @param array<string, string> $out */
    protected function flatten(array &$out, string $prefix, array $values): void
    {
        foreach ($values as $key => $value) {
            $compositeKey = $prefix === '' ? (string) $key : $prefix.'.'.$key;
            if (is_array($value)) {
                $this->flatten($out, $compositeKey, $value);
            } else {
                $out[$compositeKey] = (string) $value;
            }
        }
    }
}
```

```typescript
// Source: 01-CONTEXT.md "Vue helper t(key, params)" + UI-SPEC.md ":?param" interpolation.
// We resolve from Inertia's shared `translations` prop directly — no separate JSON bundle, no
// build-step glue. This avoids the Pitfall 8 SSR async-glob trap entirely.

import { computed } from 'vue';
import { usePage } from '@inertiajs/vue3';

type Params = Record<string, string | number>;

export function useT() {
    const page = usePage();

    const translations = computed(() => (page.props.translations ?? {}) as Record<string, string>);

    function t(key: string, params: Params = {}): string {
        const raw = translations.value[key];
        if (raw === undefined) {
            // Surface missing keys loudly in dev; silent fallback in production builds.
            if (import.meta.env.DEV) {
                console.warn(`[i18n] missing key: ${key}`);
            }
            return key;
        }
        return Object.entries(params).reduce(
            (out, [k, v]) => out.replaceAll(`:${k}`, String(v)),
            raw,
        );
    }

    return { t };
}

// Re-export a bare `t` for convenient template imports inside <script setup>.
export function t(key: string, params: Params = {}): string {
    return useT().t(key, params);
}
```

**6. Author `apps/web/tests/Feature/I18n/NoHardcodedStringsTest.php`:**

```php
<?php

declare(strict_types=1);

/*
| Source: 01-VALIDATION.md (NoHardcodedStringsTest) + UI-SPEC.md Definition of "Visually Correct" #4.
| Greps the Vue templates of pages, layouts, and components for any English literal text node
| that is NOT inside a `t()` / `t(` / `__()` call, NOT a class= attribute, NOT an HTML tag,
| NOT inside <script> blocks, NOT inside an <!-- comment -->.
*/

it('contains no hardcoded English strings in pages, layouts, or components', function (): void {
    $roots = [
        base_path('resources/js/pages'),
        base_path('resources/js/layouts'),
        base_path('resources/js/components'),
    ];

    $offenders = [];

    foreach ($roots as $root) {
        $dirIter = new RecursiveDirectoryIterator($root, RecursiveDirectoryIterator::SKIP_DOTS);
        $iter = new RecursiveIteratorIterator($dirIter);
        foreach ($iter as $file) {
            /** @var SplFileInfo $file */
            if ($file->getExtension() !== 'vue') {
                continue;
            }
            $contents = (string) file_get_contents($file->getRealPath());

            // Strip <script> / <style> / <!-- ... --> sections — only check <template>.
            $template = strstr($contents, '<template>') ?: '';
            $template = strstr($template, '</template>', true) ?: $template;

            // Remove HTML comments
            $template = preg_replace('/<!--.*?-->/s', '', $template) ?? '';

            // Find any text content NOT inside {{ ... }} or attribute values.
            // Naive: find any sequence of >=3 ASCII letters that lives between a `>` and a `<`.
            // Then assert each such match is wrapped in {{ t(...) }} / {{ __(...) }} OR is in an allowlist.
            preg_match_all('/>([^<]{3,})</', $template, $matches);

            foreach ($matches[1] as $candidate) {
                $candidate = trim($candidate);
                if ($candidate === '') {
                    continue;
                }

                // Skip pure whitespace, numbers, special chars, single characters
                if (! preg_match('/[A-Za-z]{3,}/', $candidate)) {
                    continue;
                }

                // Skip allowlist (common Vue interpolation, year math, special chars)
                if (preg_match('/^\\{\\{\\s*(t\\(|__\\(|year|.*\\$\\{).*\\}\\}/', $candidate)) {
                    continue;
                }

                // Skip if the entire content is a Vue interpolation: {{ ... }}
                if (preg_match('/^\\{\\{.*\\}\\}$/s', $candidate)) {
                    continue;
                }

                // Skip allowlisted literals (none in P1 — fail loudly otherwise).
                $offenders[] = sprintf('%s :: %s', basename((string) $file->getRealPath()), substr($candidate, 0, 60));
            }
        }
    }

    expect($offenders)
        ->toBe(
            [],
            "Hardcoded English strings found:\n  - " . implode("\n  - ", $offenders),
        );
});
```
