```vue
<script setup lang="ts">
import FullCalendar from '@fullcalendar/vue3';
import dayGridPlugin from '@fullcalendar/daygrid';
import timeGridPlugin from '@fullcalendar/timegrid';
import interactionPlugin from '@fullcalendar/interaction';
import { router } from '@inertiajs/vue3';
import { t } from '@/composables/useT';

defineProps<{ categories: App.Data.PublicCategoryData[] }>();

const calendarOptions = {
  plugins: [dayGridPlugin, timeGridPlugin, interactionPlugin],
  initialView: 'dayGridMonth',
  timeZone: 'local',  // Pitfall 11 — explicit; ISO strings carry Z suffix per CalendarEventData
  headerToolbar: {
    left: 'prev,next today',
    center: 'title',
    right: 'dayGridMonth,timeGridWeek,timeGridDay',
  },
  events: '/events/feed.json',
  buttonText: {
    today: t('events.navigation.today.label'),
    month: t('events.legend.month.label'),  // or events.toolbar
    week: t('events.legend.week.label'),
    day: t('events.legend.day.label'),
  },
  eventClick: (info: any) => {
    info.jsEvent.preventDefault();
    router.visit(info.event.extendedProps.url ?? info.event.url);
  },
};
</script>

<template>
  <PublicLayout>
    <div class="container mx-auto py-8">
      <h1 class="text-3xl font-bold mb-6">{{ t('events.page.title') }}</h1>
      <CalendarLegend />
      <FullCalendar :options="calendarOptions" />
    </div>
  </PublicLayout>
</template>
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-10-01 | Tampering | XSS via v-html bodyHtml | mitigate | 4-layer defence: (1) Tiptap editor profile pinned in 07-01 config; (2) ->profile('default') on form field in 07-05; (3) tiptap_converter()->asHTML server-side render in PublicArticleData in 07-05; (4) Pest assertion at HTTP layer in ArticleShowPageTest asserts bodyHtml does not contain <iframe> or <script> |
| T-07-10-02 | Information Disclosure | FullCalendar leaking internal event IDs in DOM | accept | CalendarEventData.id is UUID — non-sequential, non-leak; appearing in DOM is by design for FullCalendar's eventClick handler |
| T-07-10-03 | Spoofing | Inertia link prefetch loading unauthorised data | accept | Inertia prefetch follows public routes only; ArticlePolicy::view gates draft access regardless of prefetch |
| T-07-10-04 | Denial of Service | Vue v-html rendering deeply nested DOM (parser overload) | mitigate | TipTap editor caps document depth at server side (tiptap-php enforces structure); v-html paints sanitised HTML; browser native parser handles depth |
| T-07-10-05 | Information Disclosure | PublicLayout SearchBar leaking authenticated user query history via referrer | accept | Search queries are public surface; no referrer-policy beyond Laravel defaults needed in v1 |
| T-07-10-06 | Tampering | router.visit eventClick navigating to attacker URL | mitigate | url field in CalendarEventData is server-controlled (route() generation); attacker cannot inject external URLs without compromising the Event row first |
