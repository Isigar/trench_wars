```php
// apps/web/app/Console/Commands/MakeAdminCommand.php (existing — reference shape)
protected $signature = 'trenchwars:make-admin {discord_id}';
public function handle(): int
{
    $discordId = $this->argument('discord_id');
    $user = User::where('discord_id', $discordId)->first();
    if (!$user) { $this->error("User with discord_id={$discordId} not found"); return self::FAILURE; }
    $user->assignRole('super-admin');
    $this->info("Granted super-admin to {$user->username} ({$discordId})");
    return self::SUCCESS;
}
```

MakeCmsEditorCommand mirrors this shape, swap super-admin → cms-editor.

<!-- ArticlePolicy method signatures (Laravel Policy convention) -->

```php
public function viewAny(?User $user): bool { return true; }                    // public listing
public function view(?User $user, Article $article): bool {                    // public detail
    return $article->status === 'published' || $user?->can('articles.update');  // drafts only to editors
}
public function create(?User $user): bool { return $user?->can('articles.create') ?? false; }
public function update(?User $user, Article $article): bool {
    return $user?->can('articles.update') && (
        $user->id === $article->author_user_id || $user->can('articles.publish')
    );
}
public function publish(?User $user, Article $article): bool { return $user?->can('articles.publish') ?? false; }
public function delete(?User $user, Article $article): bool { return $user?->hasRole('super-admin') ?? false; }
```
