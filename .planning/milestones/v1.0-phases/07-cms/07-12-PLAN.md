use Spatie\Sitemap\Tags\Url;
public function toSitemapTag(): Url
{
    return Url::create(route('blog.show', $this->slug))
        ->setLastModificationDate($this->updated_at)
        ->setChangeFrequency(Url::CHANGE_FREQUENCY_WEEKLY)
        ->setPriority(0.7);
}

<!-- SitemapGenerateCommand (verbatim from RESEARCH Pattern 6 + Code Examples) -->

```php
use Spatie\Sitemap\Sitemap;
use App\Models\{Article, Clan, Tournament};

public function handle(): int
{
    Sitemap::create()
        ->add('/')
        ->add('/clans')
        ->add('/players')       // index only; Pitfall (privacy)
        ->add('/matches')
        ->add('/tournaments')
        ->add('/blog')
        ->add('/events')
        ->add(Article::where('status', 'published')->get())
        ->add(Clan::all())
        ->add(Tournament::where('is_public', true)->get())
        ->writeToFile(public_path('sitemap.xml'));

    $this->info('sitemap.xml written.');
    return self::SUCCESS;
}
```
