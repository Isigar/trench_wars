ArticleStatusService::transition mirrors Phase 6 D-06-04-A TournamentStatusService — accepts `?User $causer = null` (Filament admin actions omit causer), returns `Article` for fluent chaining. Captures `$from = $a->status` BEFORE update (Phase 4 D-04-04-A precedent prevents getOriginal() drift).

```php
public function transition(Article $a, string $to, ?User $causer = null): Article
{
    $from = $a->status;
    $permitted = [
        'draft' => ['scheduled', 'published'],
        'scheduled' => ['published', 'draft'],
        'published' => ['draft'],
    ];
    if (! in_array($to, $permitted[$from] ?? [], true)) {
        throw new InvalidArticleStatusTransitionException($from, $to);
    }
    $updates = ['status' => $to];
    if ($to === 'published') {
        $updates['published_at'] = now();
        $updates['scheduled_at'] = null;
    }
    $a->update($updates);
    return $a->fresh();
}
```

```php
class ArticleObserver
{
    public function created(Article $a): void
    {
        if ($a->status === 'published') {
            $this->onPublish($a);
        }
        $this->syncEvent($a);
    }

    public function updated(Article $a): void
    {
        if (! $a->wasChanged('status')) {
            $this->syncEvent($a);  // status unchanged but title/category may have changed
            return;
        }
        // Pitfall 10 in observer — only first transition to published
        if ($a->status === 'published' && $a->getOriginal('status') !== 'published') {
            $this->onPublish($a);
        }
        $this->syncEvent($a);
    }

    private function syncEvent(Article $a): void
    {
        // Phase 4 D-04-08-C precedent — is_public=false hides drafts from calendar
        $a->events()->updateOrCreate(
            ['eventable_type' => $a->getMorphClass(), 'eventable_id' => $a->id],
            [
                'is_public' => $a->status === 'published',
                'title' => $a->getTranslations('title'),
                'starts_at' => $a->published_at ?? $a->scheduled_at ?? $a->created_at,
                'ends_at' => null,  // articles are point-in-time events
            ]
        );
    }

    private function onPublish(Article $a): void
    {
        if (! $a->allow_discord_announce) return;
        $channelId = config('discord.league_announce_channel_id');
        if (empty($channelId)) return;  // defensive — no global channel configured

        DiscordOutboundMessage::create([
            'channel_id' => '',  // resolved at dispatch via config lookup (D-06-10-E precedent)
            'message_type' => 'article_announce',
            'status' => 'pending',
            'payload' => DiscordOutboundPayloadBuilder::buildArticleAnnounce($a),
            'causer_user_id' => auth()->id(),  // nullable when scheduler-driven (T-07-06-04 accept)
        ]);
    }
}
```
