<!-- ArticleResource form schema (verbatim from RESEARCH Pattern 1 + Pattern 2) -->

```php
use Filament\Forms\Components\{Section, Select, TextInput, Toggle, DateTimePicker, SpatieMediaLibraryFileUpload};
use FilamentTiptapEditor\TiptapEditor;
use FilamentTiptapEditor\Enums\TiptapOutput;

public static function form(Form $form): Form
{
    return $form->schema([
        Section::make(__('admin.article.fields.content'))->schema([
            TextInput::make('title.en')->label(__('admin.article.fields.title.label'))->required()->maxLength(200),
            TextInput::make('slug')->required()->maxLength(200)->unique(ignoreRecord: true)->disabledOn('edit'),
            Select::make('category_id')->relationship('category', 'name->en')->required()->preload(),
            TextInput::make('excerpt.en')->label(__('admin.article.fields.excerpt.label'))->maxLength(500),
            SpatieMediaLibraryFileUpload::make('hero')
                ->collection('hero')->image()->imageEditor()->responsiveImages()
                ->maxSize(5 * 1024)
                ->acceptedFileTypes(['image/jpeg', 'image/png', 'image/webp'])
                ->columnSpanFull(),
            TiptapEditor::make('body.en')
                ->label(__('admin.article.fields.body.label'))
                ->profile('default')                  // pinned in 07-01 config/filament-tiptap-editor.php
                ->output(TiptapOutput::Json)          // never Html (Pitfall 1)
                ->disk('public')->directory('article-media')
                ->maxContentWidth('5xl')
                ->required()->columnSpanFull(),
        ]),
        Section::make(__('admin.article.publication'))->schema([
            Select::make('status')
                ->options([
                    'draft' => __('cms.status.draft.label'),
                    'scheduled' => __('cms.status.scheduled.label'),
                    'published' => __('cms.status.published.label'),
                ])
                ->required()->reactive(),
            DateTimePicker::make('scheduled_at')
                ->visible(fn (callable $get) => $get('status') === 'scheduled')
                ->required(fn (callable $get) => $get('status') === 'scheduled')
                ->minDate(now()),                     // Open Question 4 + T-07-05-02 — cannot schedule in the past
            DateTimePicker::make('published_at')
                ->visible(fn (callable $get) => $get('status') === 'published')
                ->disabled(),                          // set by observer (plan 07-06)
            Toggle::make('allow_discord_announce')->default(true)
                ->helperText(__('cms.fields.allow_discord_announce.help')),
        ])->columns(2),
    ]);
}
```

```php
public static function fromModel(Article $a, ?string $locale = null): self
{
    $locale ??= app()->getLocale();
    $bodyTranslations = $a->getTranslations('body');
    $bodyJson = $bodyTranslations[$locale] ?? $bodyTranslations['en'] ?? [];

    return new self(
        id: $a->id,
        slug: $a->slug,
        title: $a->getTranslation('title', $locale, false) ?? $a->getTranslation('title', 'en', false) ?? '',
        excerpt: $a->getTranslation('excerpt', $locale, false) ?? '',
        bodyHtml: tiptap_converter()->asHTML(is_string($bodyJson) ? json_decode($bodyJson, true) : $bodyJson),
        heroThumbUrl: $a->getFirstMediaUrl('hero', 'thumb') ?: null,
        heroOgImageUrl: $a->getFirstMediaUrl('hero', 'og-image') ?: null,
        categoryName: $a->category?->getTranslation('name', $locale, false) ?? '',
        authorName: $a->author?->username,
        publishedAt: $a->published_at,
        allowDiscordAnnounce: $a->allow_discord_announce,
        url: route('blog.show', $a->slug),
    );
}
```
