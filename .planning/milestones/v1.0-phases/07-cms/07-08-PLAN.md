```php
namespace App\Services;
use App\Data\{SearchResultData, SearchResultsData};
use App\Models\{Article, Clan, Player, User};
use App\Services\PlayerPrivacyGate;
use Illuminate\Support\Facades\DB;

final class SearchService
{
    public function __construct(private PlayerPrivacyGate $privacyGate) {}

    public function search(string $q, ?User $viewer = null): SearchResultsData
    {
        $q = trim($q);
        if ($q === '') {
            return SearchResultsData::empty($q);
        }

        // plainto_tsquery sanitises stray operators (Pitfall 2) — NEVER use to_tsquery on user input
        $articles = Article::query()
            ->whereRaw("search_vector @@ plainto_tsquery('simple', ?)", [$q])
            ->where('status', 'published')
            ->orderByRaw("ts_rank(search_vector, plainto_tsquery('simple', ?)) DESC", [$q])
            ->limit(20)->get()
            ->map(fn (Article $a) => SearchResultData::fromArticle($a));

        $clans = Clan::query()
            ->whereRaw("search_vector @@ plainto_tsquery('simple', ?)", [$q])
            ->orderByRaw("ts_rank(search_vector, plainto_tsquery('simple', ?)) DESC", [$q])
            ->limit(20)->get()
            ->map(fn (Clan $c) => SearchResultData::fromClan($c));

        $players = Player::query()
            ->whereRaw("search_vector @@ plainto_tsquery('simple', ?)", [$q])
            ->orderByRaw("ts_rank(search_vector, plainto_tsquery('simple', ?)) DESC", [$q])
            ->limit(20)->get()
            ->filter(fn (Player $p) => $this->privacyGate->canShowInSearch($p, $viewer))
            ->map(fn (Player $p) => SearchResultData::fromPlayer($p, $this->privacyGate, $viewer));

        return SearchResultsData::fromQuery($q, $articles->all(), $clans->all(), $players->values()->all());
    }
}
```

```php
public function canShowInSearch(Player $player, ?User $viewer): bool
{
    // self-bypass
    if ($viewer && $viewer->players()->where('id', $player->id)->exists()) {
        return true;
    }
    $tier = $player->privacy?->show_to ?? 'public';
    return match ($tier) {
        'public' => true,
        'community' => $viewer !== null,
        'clan' => $viewer && $this->shareActiveClan($player, $viewer),
        'private' => false,
        default => false,
    };
}

private function shareActiveClan(Player $player, User $viewer): bool
{
    $viewerPlayerIds = $viewer->players()->pluck('id');
    return $player->activeClanMembership && \App\Models\ClanMembership::query()
        ->whereIn('player_id', $viewerPlayerIds)
        ->where('clan_id', $player->activeClanMembership->clan_id)
        ->whereNull('left_at')
        ->exists();
}
```
