Filament HasWizard pattern (verbatim from RESEARCH Pattern 6 + Context7 panels/resources/creating-records):
```php
use Filament\Resources\Pages\CreateRecord;
use Filament\Forms\Components\Wizard\Step;

class CreateMatch extends CreateRecord
{
    use CreateRecord\Concerns\HasWizard;

    protected static string $resource = MatchResource::class;

    protected function getSteps(): array
    {
        return [
            Step::make(__('admin.match.wizard.step_type'))->schema([...]),
            Step::make(__('admin.match.wizard.step_schedule'))->schema([...]),
            Step::make(__('admin.match.wizard.step_review'))->schema([...]),
        ];
    }

    protected function handleRecordCreation(array $data): Model
    {
        return DB::transaction(function () use ($data) {
            $data['status'] = 'open';  // SC-1: signups open automatically
            $match = static::getModel()::create($data);
            app(MatchSlotMaterialiserService::class)->materialise($match);
            return $match;
        });
    }
}
```

MatchResultService (NEW — pattern follows ClanInviteService):
```
public function upsert(MatchModel $match, array $data, User $causer): MatchResult
{
    return DB::transaction(function () use ($match, $data, $causer): MatchResult {
        $result = MatchResult::updateOrCreate(
            ['match_id' => $match->id],
            [
                'winner_clan_id' => $data['winner_clan_id'] ?? null,
                'allies_score' => $data['allies_score'] ?? null,
                'axis_score' => $data['axis_score'] ?? null,
                'notes' => $data['notes'] ?? null,
                'recorded_by_user_id' => $causer->id,
                'recorded_at' => $data['recorded_at'] ?? now(),
            ],
        );

        // Atomic status transition to 'played' on first result write (skip if already 'played')
        if ($match->status !== 'played') {
            app(MatchStatusService::class)->transition($match, 'played', $causer);
        }

        return $result;
    });
}
```
