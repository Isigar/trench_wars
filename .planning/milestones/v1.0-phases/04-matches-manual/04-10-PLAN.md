```
class MatchShowController
{
    public function __invoke(Match $match, PlayerPrivacyGate $gate, Request $req): Response
    {
        $match->load([
            'gameMatchType', 'gameMatchType.roleLimits',
            'slots.occupantUser.player.privacy',
            'slots.occupantUser.activeClanMembership.clan',
            'accessRules.clanTag', 'result.mvps.player',
        ]);

        if (!$match->is_public && $match->organiser_user_id !== $req->user()?->id) {
            abort(404);
        }

        $viewer = $req->user();
        $roleGroups = $match->slots->groupBy('game_role_id')->map(function ($slots) use ($gate, $viewer) {
            $role = $slots->first()->role;
            $occupantDtos = $slots->map(fn ($slot) => PublicMatchOccupantData::fromMatchSlot($slot, $viewer, $gate));
            return ['role' => GameRoleData::fromModel($role), 'slots' => $occupantDtos->values()->all()];
        });

        return Inertia::render('Matches/Show', [
            'match' => PublicMatchData::fromModel($match),
            'roleGroups' => $roleGroups->values()->all(),
            'signupAllowed' => $this->computeSignupAllowed($match, $viewer),
            'viewerSlotId' => $this->findViewerSlot($match, $viewer)?->id,
        ]);
    }
}
```

Calendar controller (SC-3 first half — derived from Pattern 7 calendar pseudocode):

```
class MatchCalendarController
{
    public function __invoke(Request $request): Response
    {
        $request->validate([
            'date_from' => 'nullable|date',
            'date_to' => 'nullable|date',
            'tag' => 'nullable|string|alpha_dash|max:32',
            'status' => 'nullable|in:open,locked,played',
        ]);

        $query = Match::with(['gameMatchType', 'event', 'slots'])
            ->where('is_public', true)
            ->whereIn('status', ['open', 'locked', 'played'])  // exclude draft + cancelled
            ->whereDate('scheduled_at', '>=', $request->input('date_from', today()->toDateString()));

        if ($request->has('date_to')) {
            $query->whereDate('scheduled_at', '<=', $request->input('date_to'));
        }
        if ($request->has('tag')) {
            $tag = ClanTag::where('slug', $request->input('tag'))->firstOrFail();
            $query->whereHas('accessRules', fn ($q) => $q->where('clan_tag_id', $tag->id));
        }
        if ($request->has('status')) {
            $query->where('status', $request->input('status'));
        }

        $matches = $query->orderBy('scheduled_at')->paginate(20);

        return Inertia::render('Matches/Index', [
            'matches' => PublicMatchData::collect($matches->getCollection()),
            'pagination' => [
                'currentPage' => $matches->currentPage(),
                'lastPage' => $matches->lastPage(),
                'total' => $matches->total(),
                'perPage' => $matches->perPage(),
            ],
            'activeFilters' => [
                'dateFrom' => $request->input('date_from'),
                'dateTo' => $request->input('date_to'),
                'tag' => $request->input('tag'),
                'status' => $request->input('status'),
            ],
        ]);
    }
}
```

Signup controller (the auth POST handler):

```
class MatchSignupController extends Controller
{
    public function store(MatchSignupRequest $request, Match $match, MatchSignupService $service): RedirectResponse
    {
        try {
            $slot = $service->signup($match, $request->user(), GameRole::findOrFail($request->validated()['game_role_id']));
            return back()->with('success', __('matches.signup.success', ['role' => $slot->role->display_name['en']]));
        } catch (CapacityExceededException $e) {
            return back()->withErrors(['game_role_id' => $e->getMessage()])->status(422);
        } catch (TagRestrictedException $e) {
            return back()->withErrors(['general' => $e->getMessage()])->status(422);
        } catch (MatchNotOpenException $e) {
            return back()->withErrors(['general' => $e->getMessage()])->status(422);
        } catch (AlreadySignedUpException $e) {
            return back()->withErrors(['general' => $e->getMessage()])->status(422);
        }
    }

    public function destroy(Request $request, Match $match, MatchSlot $slot): RedirectResponse
    {
        abort_unless($slot->occupant_user_id === $request->user()->id, 403);
        $slot->update(['occupant_user_id' => null, 'confirmed_at' => null]);
        return back()->with('success', __('matches.signup.cancelled'));
    }
}
```
