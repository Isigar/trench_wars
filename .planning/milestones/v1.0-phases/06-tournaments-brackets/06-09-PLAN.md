```php
declare(strict_types=1);
namespace App\Services;

use App\Models\Tournament;
use App\Services\Standings\DoubleEliminationStandingsCalculator;
use App\Services\Standings\RoundRobinStandingsCalculator;
use App\Services\Standings\SingleEliminationStandingsCalculator;
use App\Services\Standings\SwissStandingsCalculator;
use Illuminate\Support\Facades\DB;

final class StandingsCalculatorService
{
    public function __construct(
        private readonly SingleEliminationStandingsCalculator $singleElim,
        private readonly DoubleEliminationStandingsCalculator $doubleElim,
        private readonly RoundRobinStandingsCalculator $roundRobin,
        private readonly SwissStandingsCalculator $swiss,
    ) {}

    public function recalculate(Tournament $tournament): void
    {
        DB::transaction(function () use ($tournament): void {
            // Pitfall 6 — serialise on parent tournament row
            Tournament::lockForUpdate()->find($tournament->id);

            // Wipe + recompute (simpler than upsert; small tables)
            $tournament->standings()->delete();

            $strategy = match ($tournament->format) {
                'single_elimination' => $this->singleElim,
                'double_elimination' => $this->doubleElim,
                'round_robin' => $this->roundRobin,
                'swiss' => $this->swiss,
            };

            $strategy->compute($tournament);
        });
    }
}
```

SingleEliminationStandingsCalculator (placement from bracket position):

```php
<?php
declare(strict_types=1);
namespace App\Services\Standings;

use App\Models\Tournament;
use App\Models\TournamentBracket;
use App\Models\TournamentStanding;
use App\Models\MatchResult;

final class SingleEliminationStandingsCalculator implements StandingsCalculatorStrategy
{
    public function compute(Tournament $tournament): void
    {
        $stage = $tournament->stages()->where('type', 'elim')->first();
        if ($stage === null) {
            return;
        }

        // Compute per-participant wins/losses
        $participants = $tournament->participants()->whereIn('status', ['active','withdrawn','disqualified'])->get();
        foreach ($participants as $participant) {
            $wins = 0;
            $losses = 0;
            $matchIds = $stage->brackets()
                ->where(function ($q) use ($participant) {
                    $q->where('participant_a_id', $participant->id)
                        ->orWhere('participant_b_id', $participant->id);
                })
                ->whereNotNull('match_id')
                ->pluck('match_id');
            $results = MatchResult::whereIn('match_id', $matchIds)->whereNotNull('winner_clan_id')->get();
            foreach ($results as $r) {
                if ($r->winner_clan_id === $participant->clan_id) $wins++;
                else $losses++;
            }

            // Placement: derived from elimination round
            $placement = $this->derivePlacement($stage, $participant);

            TournamentStanding::create([
                'tournament_id' => $tournament->id,
                'tournament_stage_id' => $stage->id,
                'participant_id' => $participant->id,
                'wins' => $wins,
                'losses' => $losses,
                'draws' => 0,           // single-elim has no draws
                'points' => $wins,      // 1 per win
                'tiebreak_score' => 0,  // not used for single-elim
                'rank' => $placement,
            ]);
        }
    }

    /**
     * Find the bracket where the participant was eliminated; placement is derived from that bracket's round number.
     * winner of final (round N) = placement 1; loser of final = placement 2; semi-final losers = placements 3-4
     * (assigned by seed-based tiebreak); QF losers = 5-8 (by seed); etc.
     */
    private function derivePlacement($stage, $participant): ?int
    {
        $finalRound = $stage->brackets()->max('round_number');
        $finalBracket = $stage->brackets()->where('round_number', $finalRound)->first();

        // Tournament winner
        if ($finalBracket?->winner_participant_id === $participant->id) {
            return 1;
        }
        // Tournament runner-up = final loser
        if ($finalBracket?->participant_a_id === $participant->id || $finalBracket?->participant_b_id === $participant->id) {
            return 2;
        }

        // Find which round the participant was eliminated in
        $lastBracket = $stage->brackets()
            ->where(function ($q) use ($participant) {
                $q->where('participant_a_id', $participant->id)
                    ->orWhere('participant_b_id', $participant->id);
            })
            ->orderByDesc('round_number')
            ->first();

        if ($lastBracket === null) {
            return null;  // not yet placed
        }

        // Placement range for losers in round R: [2^(finalRound - R) + 1, 2^(finalRound - R + 1)]
        // e.g. for 8-team tournament (finalRound=3):
        //   Lost in round 1 (R=1, finalRound=3): placements 5-8 → 2^2+1 to 2^3 = 5-8
        //   Lost in round 2 (R=2): placements 3-4 → 2^1+1 to 2^2 = 3-4
        //   Lost in round 3 (R=3): placement 2
        $roundEliminated = $lastBracket->round_number;
        $rangeStart = (2 ** ($finalRound - $roundEliminated)) + 1;
        // v1: assign the min of the range (all losers in same round share placement); admin sees [3, 5, 7] for an 8-team tournament's eliminated participants
        // Phase 9 polish: tiebreak by seed within the range to get distinct placements
        return $rangeStart;
    }
}
```

RoundRobinStandingsCalculator (FIFA points + tiebreakers):

```php
<?php
declare(strict_types=1);
namespace App\Services\Standings;

use App\Models\Tournament;
use App\Models\TournamentStanding;
use App\Models\MatchResult;
use App\Models\TournamentParticipant;

final class RoundRobinStandingsCalculator implements StandingsCalculatorStrategy
{
    public function compute(Tournament $tournament): void
    {
        $stage = $tournament->stages()->where('type', 'group')->first();
        if ($stage === null) return;

        // Resolve points scheme (admin override via tournament.settings)
        $pointsWin = $tournament->settings['roundrobin_points_per_win'] ?? 3;
        $pointsDraw = $tournament->settings['roundrobin_points_per_draw'] ?? 1;
        $pointsLoss = $tournament->settings['roundrobin_points_per_loss'] ?? 0;

        // Build per-participant aggregate
        $stats = [];
        $participants = $tournament->participants()->whereIn('status', ['active','withdrawn','disqualified'])->get();
        $matchIds = $stage->brackets()->whereNotNull('match_id')->pluck('match_id');
        $results = MatchResult::whereIn('match_id', $matchIds)->get()->keyBy('match_id');

        foreach ($participants as $p) {
            $stats[$p->id] = ['wins' => 0, 'losses' => 0, 'draws' => 0, 'point_diff' => 0, 'head_to_head' => []];
        }

        foreach ($stage->brackets()->whereNotNull('match_id')->get() as $bracket) {
            $result = $results->get($bracket->match_id);
            if ($result === null) continue;
            $pAId = $bracket->participant_a_id;
            $pBId = $bracket->participant_b_id;
            if (!$pAId || !$pBId) continue;

            $aClanId = TournamentParticipant::find($pAId)?->clan_id;
            $bClanId = TournamentParticipant::find($pBId)?->clan_id;

            if ($result->winner_clan_id === null) {
                $stats[$pAId]['draws']++;
                $stats[$pBId]['draws']++;
            } elseif ($result->winner_clan_id === $aClanId) {
                $stats[$pAId]['wins']++;
                $stats[$pBId]['losses']++;
                $stats[$pAId]['head_to_head'][$pBId] = ($stats[$pAId]['head_to_head'][$pBId] ?? 0) + 1;
            } else {
                $stats[$pBId]['wins']++;
                $stats[$pAId]['losses']++;
                $stats[$pBId]['head_to_head'][$pAId] = ($stats[$pBId]['head_to_head'][$pAId] ?? 0) + 1;
            }

            $stats[$pAId]['point_diff'] += ($result->a_score ?? 0) - ($result->b_score ?? 0);
            $stats[$pBId]['point_diff'] += ($result->b_score ?? 0) - ($result->a_score ?? 0);
        }

        // Compute points + sort
        $rows = [];
        foreach ($participants as $p) {
            $s = $stats[$p->id];
            $points = $s['wins'] * $pointsWin + $s['draws'] * $pointsDraw + $s['losses'] * $pointsLoss;
            $rows[] = [
                'participant' => $p,
                'wins' => $s['wins'],
                'losses' => $s['losses'],
                'draws' => $s['draws'],
                'points' => $points,
                'point_diff' => $s['point_diff'],
                'head_to_head' => $s['head_to_head'],
            ];
        }

        // Sort by (points DESC, head_to_head wins among tied participants DESC, point_diff DESC, seed ASC)
        usort($rows, function ($a, $b) use (&$rows) {
            if ($a['points'] !== $b['points']) return $b['points'] - $a['points'];
            // Head-to-head: count wins among the tied subset (v1: simple direct head-to-head; full mini-table is Phase 9)
            $aWins = $a['head_to_head'][$b['participant']->id] ?? 0;
            $bWins = $b['head_to_head'][$a['participant']->id] ?? 0;
            if ($aWins !== $bWins) return $bWins - $aWins;
            if ($a['point_diff'] !== $b['point_diff']) return $b['point_diff'] - $a['point_diff'];
            return ($a['participant']->seed ?? PHP_INT_MAX) - ($b['participant']->seed ?? PHP_INT_MAX);
        });

        // Write standings rows with rank
        foreach ($rows as $rank => $row) {
            TournamentStanding::create([
                'tournament_id' => $tournament->id,
                'tournament_stage_id' => $stage->id,
                'participant_id' => $row['participant']->id,
                'wins' => $row['wins'],
                'losses' => $row['losses'],
                'draws' => $row['draws'],
                'points' => $row['points'],
                'tiebreak_score' => $row['point_diff'],
                'rank' => $rank + 1,
            ]);
        }
    }
}
```

SwissStandingsCalculator: similar pattern but points 1/0.5/0 and tiebreak_score = Buchholz (sum of opponents' current scores).

```php
final class SwissStandingsCalculator implements StandingsCalculatorStrategy
{
    public function compute(Tournament $tournament): void
    {
        $stages = $tournament->stages()->where('type', 'swiss-round')->get();
        if ($stages->isEmpty()) return;

        // First pass: compute raw points per participant (1/0.5/0)
        $participants = $tournament->participants()->whereIn('status', ['active','withdrawn','disqualified'])->get();
        $pointsByParticipant = [];
        $opponentsByParticipant = [];
        foreach ($participants as $p) {
            $pointsByParticipant[$p->id] = 0.0;
            $opponentsByParticipant[$p->id] = [];
        }

        $matchIds = collect();
        foreach ($stages as $stage) {
            $matchIds = $matchIds->merge($stage->brackets()->whereNotNull('match_id')->pluck('match_id'));
        }
        $results = MatchResult::whereIn('match_id', $matchIds)->get()->keyBy('match_id');

        foreach ($stages as $stage) {
            foreach ($stage->brackets()->get() as $bracket) {
                $pAId = $bracket->participant_a_id;
                $pBId = $bracket->participant_b_id;

                // Handle bye (winner already set, no match)
                if ($pAId && !$pBId && $bracket->winner_participant_id === $pAId) {
                    $pointsByParticipant[$pAId] += 1.0;  // bye worth 1 point (RESEARCH Pattern 5)
                    continue;
                }
                if (!$pAId || !$pBId) continue;

                $opponentsByParticipant[$pAId][] = $pBId;
                $opponentsByParticipant[$pBId][] = $pAId;

                if (!$bracket->match_id) continue;
                $result = $results->get($bracket->match_id);
                if ($result === null) continue;

                $aClanId = TournamentParticipant::find($pAId)?->clan_id;
                $bClanId = TournamentParticipant::find($pBId)?->clan_id;

                if ($result->winner_clan_id === null) {
                    $pointsByParticipant[$pAId] += 0.5;
                    $pointsByParticipant[$pBId] += 0.5;
                } elseif ($result->winner_clan_id === $aClanId) {
                    $pointsByParticipant[$pAId] += 1.0;
                } else {
                    $pointsByParticipant[$pBId] += 1.0;
                }
            }
        }

        // Second pass: compute Buchholz (sum of each participant's opponents' current scores)
        $buchholzByParticipant = [];
        foreach ($participants as $p) {
            $sum = 0.0;
            foreach ($opponentsByParticipant[$p->id] as $oppId) {
                $sum += $pointsByParticipant[$oppId] ?? 0.0;
            }
            $buchholzByParticipant[$p->id] = $sum;
        }

        // Build rows + count W/L/D per participant
        $rows = [];
        foreach ($participants as $p) {
            $wins = 0; $losses = 0; $draws = 0;
            foreach ($stages as $stage) {
                foreach ($stage->brackets()->get() as $bracket) {
                    if ($bracket->participant_a_id === $p->id || $bracket->participant_b_id === $p->id) {
                        if (!$bracket->match_id) continue;
                        $result = $results->get($bracket->match_id);
                        if (!$result) continue;
                        if ($result->winner_clan_id === null) {
                            $draws++;
                        } elseif ($result->winner_clan_id === $p->clan_id) {
                            $wins++;
                        } else {
                            $losses++;
                        }
                    }
                }
            }
            $rows[] = [
                'participant' => $p,
                'wins' => $wins,
                'losses' => $losses,
                'draws' => $draws,
                'points' => $pointsByParticipant[$p->id],
                'buchholz' => $buchholzByParticipant[$p->id],
            ];
        }

        // Sort by (points DESC, buchholz DESC, seed ASC)
        usort($rows, function ($a, $b) {
            if ((float)$a['points'] !== (float)$b['points']) return $b['points'] <=> $a['points'];
            if ((float)$a['buchholz'] !== (float)$b['buchholz']) return $b['buchholz'] <=> $a['buchholz'];
            return ($a['participant']->seed ?? PHP_INT_MAX) - ($b['participant']->seed ?? PHP_INT_MAX);
        });

        // Write standings (use first swiss stage as stage_id; all swiss rounds aggregate into one standings table)
        $primaryStage = $stages->first();
        foreach ($rows as $rank => $row) {
            TournamentStanding::create([
                'tournament_id' => $tournament->id,
                'tournament_stage_id' => $primaryStage->id,
                'participant_id' => $row['participant']->id,
                'wins' => $row['wins'],
                'losses' => $row['losses'],
                'draws' => $row['draws'],
                'points' => $row['points'],
                'tiebreak_score' => $row['buchholz'],
                'rank' => $rank + 1,
            ]);
        }
    }
}
```
