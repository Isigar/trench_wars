1. **BracketGeneratorStrategy** (interface) — pure void contract: `generate(Tournament, Collection): void`. The 3 remaining strategies (double-elim, round-robin, swiss) land in plan 06-07.

2. **BracketGeneratorService** (front-door) — match()-based dispatch on `tournament.format`; throws `BracketsAlreadyGeneratedException` (plan 06-04) if stages exist; runs the chosen strategy inside DB::transaction. This is the SC-1 admin-clicks-Start entry point.

3. **SingleEliminationGenerator** — implements RESEARCH Pattern 3 verbatim: inner_outer seeding for sizes 4/8/16; bye distribution to top seeds; two-pass insert pattern (first pass writes brackets with null advances_to; second pass UPDATEs advances_to using a (round_number, position) lookup map). Pitfall 2 mitigation: use `(int) ceil($currentPosition / 2)` for the advances_to target position.

4. **BracketMatchMaterialiserService** — the bracket → GameMatch bridge. Row-locked + idempotent per Pitfall 4.

```php
    {
        if ($tournament->stages()->exists()) {
            throw new BracketsAlreadyGeneratedException(__('tournaments.errors.brackets_already_generated'));
        }

        $participants = $tournament->participants()
            ->where('status', 'active')
            ->orderBy('seed')
            ->get();

        $strategy = match ($tournament->format) {
            'single_elimination' => $this->singleElim,
            'double_elimination' => $this->doubleElim,
            'round_robin' => $this->roundRobin,
            'swiss' => $this->swiss,
        };

        DB::transaction(fn () => $strategy->generate($tournament, $participants));
    }
}
```

```php
<?php
declare(strict_types=1);
namespace App\Services\Brackets;

use App\Models\Tournament;
use App\Models\TournamentBracket;
use App\Models\TournamentStage;
use Illuminate\Support\Collection;

final class SingleEliminationGenerator implements BracketGeneratorStrategy
{
    /**
     * Inner_outer pre-computed orderings (RESEARCH Pattern 3; cross-referenced against brackets-manager.js Context7 docs).
     * Indexed by bracket size (must be power of 2: 4, 8, 16, 32, 64). Each value is the 1-indexed seed order
     * for round-1 pairings: pair (orderings[0], orderings[1]), (orderings[2], orderings[3]), ...
     *
     * @var array<int, list<int>>
     */
    private const INNER_OUTER_ORDERINGS = [
        2 => [1, 2],
        4 => [1, 4, 2, 3],
        8 => [1, 8, 4, 5, 2, 7, 3, 6],
        16 => [1, 16, 8, 9, 4, 13, 5, 12, 2, 15, 7, 10, 3, 14, 6, 11],
        32 => [
            1, 32, 16, 17, 8, 25, 9, 24, 4, 29, 13, 20, 5, 28, 12, 21,
            2, 31, 15, 18, 7, 26, 10, 23, 3, 30, 14, 19, 6, 27, 11, 22,
        ],
        64 => [
            // 64 entries via the recursive doubling formula — generated at runtime via computeInnerOuter()
            // OR hardcoded for performance. Recommendation: hardcode the 32-element case + recursive compute for 64+.
        ],
    ];

    public function generate(Tournament $tournament, Collection $orderedParticipants): void
    {
        $n = $orderedParticipants->count();
        if ($n < 2) {
            throw new \InvalidArgumentException(__('tournaments.errors.insufficient_participants'));  // i18n key needed in 06-01 OR per-plan add
        }

        $bracketSize = 2 ** (int) ceil(log($n, 2));
        $byeCount = $bracketSize - $n;
        $totalRounds = (int) log($bracketSize, 2);

        // Build the participant array padded with nulls for byes (top seeds get the byes per Brackets Ninja)
        $ordering = self::INNER_OUTER_ORDERINGS[$bracketSize] ?? $this->computeInnerOuter($bracketSize);
        $participantsByPos = [];
        foreach ($ordering as $i => $seedPosition) {
            // $seedPosition is 1-indexed seed; orderedParticipants is 0-indexed by ->get() collection
            $participantsByPos[$i] = $orderedParticipants->get($seedPosition - 1); // null if seed > N (bye slot)
        }

        // Create the single stage
        $stage = TournamentStage::create([
            'tournament_id' => $tournament->id,
            'type' => 'elim',
            'ordinal' => 1,
            'name' => null,
            'settings' => null,
        ]);

        // Two-pass insert: pass 1 — create all brackets with null advances_to; lookup map by (round, position)
        /** @var array<int, array<int, TournamentBracket>> $byRoundPosition */
        $byRoundPosition = [];

        for ($r = 1; $r <= $totalRounds; $r++) {
            $bracketsInRound = (int) ($bracketSize / (2 ** $r));
            for ($p = 1; $p <= $bracketsInRound; $p++) {
                $participantA = null;
                $participantB = null;
                $winnerParticipantId = null;

                if ($r === 1) {
                    // Round 1: pair from the ordering array
                    $pairIndex = ($p - 1) * 2;
                    $participantA = $participantsByPos[$pairIndex] ?? null;
                    $participantB = $participantsByPos[$pairIndex + 1] ?? null;

                    // Bye logic: if B is null and A is not null, A wins by bye
                    if ($participantB === null && $participantA !== null) {
                        $winnerParticipantId = $participantA->id;
                    }
                }

                $bracket = TournamentBracket::create([
                    'tournament_stage_id' => $stage->id,
                    'round_number' => $r,
                    'position' => $p,
                    'participant_a_id' => $participantA?->id,
                    'participant_b_id' => $participantB?->id,
                    'winner_participant_id' => $winnerParticipantId,
                    'match_id' => null,
                    'advances_to_bracket_id' => null,  // pass 2 fills this
                    'loser_advances_to_bracket_id' => null,
                ]);

                $byRoundPosition[$r][$p] = $bracket;
            }
        }

        // Pass 2 — UPDATE advances_to_bracket_id (Pitfall 2: use (int) ceil($p / 2))
        for ($r = 1; $r < $totalRounds; $r++) {
            foreach ($byRoundPosition[$r] as $p => $bracket) {
                $nextRoundPosition = (int) ceil($p / 2);  // Pitfall 2 mitigation
                $nextBracket = $byRoundPosition[$r + 1][$nextRoundPosition];
                $bracket->update(['advances_to_bracket_id' => $nextBracket->id]);

                // If the bye-winner is in round 1, propagate to round 2 immediately
                if ($r === 1 && $bracket->winner_participant_id !== null) {
                    $slot = $p % 2 === 1 ? 'participant_a_id' : 'participant_b_id';
                    $nextBracket->update([$slot => $bracket->winner_participant_id]);
                }
            }
        }
    }

    /**
     * Recursive inner_outer ordering computation for power-of-2 sizes > 32.
     * Reference: brackets-manager.js inner_outer seeding algorithm.
     * @return list<int>
     */
    private function computeInnerOuter(int $size): array
    {
        if ($size === 2) return [1, 2];
        $half = $size / 2;
        $top = $this->computeInnerOuter($half);
        $bottom = array_map(fn ($x) => $size + 1 - $x, $top);
        $result = [];
        foreach ($top as $i => $val) {
            $result[] = $val;
            $result[] = $bottom[$i];
        }
        return $result;
    }
}
```

```php
<?php
declare(strict_types=1);
namespace App\Services;

use App\Models\GameMatch;
use App\Models\Tournament;
use App\Models\TournamentBracket;
use Illuminate\Support\Facades\DB;

final class BracketMatchMaterialiserService
{
    public function __construct(
        private readonly MatchSlotMaterialiserService $slotMaterialiser,
    ) {}

    /**
     * Materialise GameMatch + slot grid for every round-1 bracket of this tournament.
     * Called from Filament Start action (plan 06-11) after BracketGeneratorService::generate().
     */
    public function materialiseFirstRound(Tournament $tournament): void
    {
        // Find the first stage's round-1 brackets that have BOTH participants set (skip byes)
        $brackets = TournamentBracket::query()
            ->whereIn('tournament_stage_id', $tournament->stages()->pluck('id'))
            ->where('round_number', 1)
            ->whereNotNull('participant_a_id')
            ->whereNotNull('participant_b_id')
            ->whereNull('match_id')
            ->get();

        foreach ($brackets as $bracket) {
            $this->materialiseFor($bracket, $tournament);
        }
    }

    /**
     * Materialise GameMatch for a single bracket. Idempotent: if bracket already has match_id, returns the existing match.
     * Row-locked per Pitfall 4 (concurrent admin clicks on "Materialise next round" would otherwise spawn duplicate matches).
     */
    public function materialiseFor(TournamentBracket $bracket, ?Tournament $tournament = null): GameMatch
    {
        return DB::transaction(function () use ($bracket, $tournament): GameMatch {
            $locked = TournamentBracket::lockForUpdate()->find($bracket->id);
            if ($locked->match_id !== null) {
                return $locked->match;  // idempotent return
            }

            $t = $tournament ?? $locked->stage->tournament;

            $match = GameMatch::create([
                'organiser_user_id' => $t->organiser_user_id,
                'host_clan_id' => null,                // A10 LOCKED — bracket matches have no host clan
                'game_match_type_id' => $t->default_game_match_type_id,
                'status' => 'open',                    // signups open automatically (D-04 plan 04-04 transition matrix accepts draft→open or direct create with open)
                'scheduled_start_at' => $t->starts_at, // tournament-level default; admin overrides per-bracket later
                'scheduled_end_at' => null,
                'title' => $t->getTranslations('title'),
                'is_public' => $t->is_public,
            ]);

            // Spawn the slot grid via Phase 4 MatchSlotMaterialiserService
            $this->slotMaterialiser->materialise($match);

            $locked->update(['match_id' => $match->id]);

            return $match;
        });
    }
}
```
