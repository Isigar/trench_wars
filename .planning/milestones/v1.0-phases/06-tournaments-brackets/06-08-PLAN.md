```php
<?php
declare(strict_types=1);
namespace App\Services;

use App\Exceptions\BracketWinnerNotParticipantException;
use App\Models\DiscordOutboundMessage;
use App\Models\MatchResult;
use App\Models\Tournament;
use App\Models\TournamentBracket;
use App\Models\TournamentParticipant;
use App\Support\DiscordOutboundPayloadBuilder;
use Illuminate\Support\Facades\DB;

final class BracketAdvancementService
{
    public function __construct(
        private readonly TournamentStatusService $statusService,
        private readonly DiscordOutboundPayloadBuilder $payloadBuilder,
        // StandingsCalculatorService is injected here via app() to break circular dependency
        // — StandingsCalculatorService may need to read brackets, BracketAdvancementService needs to call recalculate.
        // Both are auto-resolved; app() lookup avoids constructor cycle.
    ) {}

    public function advance(MatchResult $result): void
    {
        if ($result->winner_clan_id === null) {
            return;  // draw — no advancement
        }

        // Find bracket linked to this match
        $bracket = TournamentBracket::where('match_id', $result->match_id)->first();
        if ($bracket === null) {
            return;  // non-tournament match — no advancement
        }

        // Find tournament + participant
        $tournament = $bracket->stage->tournament;
        $winnerParticipant = TournamentParticipant::where('tournament_id', $tournament->id)
            ->where('clan_id', $result->winner_clan_id)
            ->first();

        if ($winnerParticipant === null) {
            throw new BracketWinnerNotParticipantException(__('tournaments.errors.winner_not_participant'));
        }

        // Resolve loser participant if both slots have participants (for double-elim loser_advances_to)
        $loserParticipant = null;
        if ($bracket->participant_a_id !== null && $bracket->participant_b_id !== null) {
            $loserParticipantId = $bracket->participant_a_id === $winnerParticipant->id
                ? $bracket->participant_b_id
                : $bracket->participant_a_id;
            $loserParticipant = TournamentParticipant::find($loserParticipantId);
        }

        DB::transaction(function () use ($bracket, $winnerParticipant, $loserParticipant, $tournament): void {
            // Pitfall 6 — serialise standings recalc on tournament-row lock
            Tournament::lockForUpdate()->find($tournament->id);

            // 1. Write winner on this bracket
            $bracket->update(['winner_participant_id' => $winnerParticipant->id]);

            // 2. Propagate winner to next bracket via advances_to_bracket_id
            if ($bracket->advances_to_bracket_id !== null) {
                $next = TournamentBracket::lockForUpdate()->find($bracket->advances_to_bracket_id);
                $slot = $this->resolveSlot($bracket->position);
                $next->update(["participant_{$slot}_id" => $winnerParticipant->id]);
            }

            // 3. Double-elim: propagate loser to loser_advances_to_bracket_id
            if ($bracket->loser_advances_to_bracket_id !== null && $loserParticipant !== null) {
                $lNext = TournamentBracket::lockForUpdate()->find($bracket->loser_advances_to_bracket_id);
                $lSlot = $this->resolveSlot($bracket->position);
                $lNext->update(["participant_{$lSlot}_id" => $loserParticipant->id]);
            }

            // 4. Grand final reset: if this bracket is in grand-final stage AND reset is enabled AND W-winner lost
            if ($bracket->stage->type === 'grand-final' && ($bracket->stage->settings['grand_final_reset'] ?? false)) {
                $wWinner = $this->findStageWinner($tournament, 'winners-bracket');
                if ($wWinner !== null && $winnerParticipant->id !== $wWinner->id && $bracket->position === 1) {
                    // Create the reset match in the same stage (ordinal=2 row)
                    TournamentBracket::create([
                        'tournament_stage_id' => $bracket->stage->id,
                        'round_number' => 2,
                        'position' => 1,
                        'participant_a_id' => $wWinner->id,
                        'participant_b_id' => $winnerParticipant->id,
                        // No advances_to — terminal
                    ]);
                }
            }

            // 5. Trigger standings recalc (lazy via app() to avoid circular DI)
            app(StandingsCalculatorService::class)->recalculate($tournament);

            // 6. Enqueue Discord bracket_result_announce
            DiscordOutboundMessage::create([
                'message_type' => 'bracket_result_announce',
                'status' => 'pending',
                'channel_id' => null,  // resolved at dispatch time by render service
                'payload' => $this->payloadBuilder->buildBracketResult($bracket->fresh()),
                'causer_user_id' => auth()->id(),
            ]);

            // 7. Tournament completion detection
            if ($this->allBracketsComplete($tournament)) {
                $this->assignFinalPlacements($tournament);
                $this->statusService->transition($tournament, 'completed', $result->causer);  // causer fallback to MatchResult.causer if exists; else auth
            }
        });
    }

    private function resolveSlot(int $fromPosition): string
    {
        // RESEARCH Pattern 7: odd from-position → slot 'a'; even → slot 'b'
        return $fromPosition % 2 === 1 ? 'a' : 'b';
    }

    private function findStageWinner(Tournament $tournament, string $stageType): ?TournamentParticipant
    {
        $stage = $tournament->stages()->where('type', $stageType)->first();
        if ($stage === null) return null;
        // Final bracket = highest round_number, position 1
        $finalBracket = $stage->brackets()->orderByDesc('round_number')->orderBy('position')->first();
        return $finalBracket?->winnerParticipant;
    }

    private function allBracketsComplete(Tournament $tournament): bool
    {
        // For single-elim / double-elim / swiss: check if every bracket has winner_participant_id set
        // For round-robin: same check — every match must have a winner (or draw flag)
        // For double-elim with reset: the LAST bracket (grand-final ordinal=2 if it exists, else ordinal=1) must have a winner

        return ! TournamentBracket::query()
            ->whereIn('tournament_stage_id', $tournament->stages()->pluck('id'))
            ->whereNull('winner_participant_id')
            // Skip brackets with a null match_id (not yet materialised) — only count brackets that have spawned a match
            ->whereNotNull('match_id')
            ->exists()
            // AND ensure at least one bracket has a match_id (avoid completing a tournament that never started)
            && TournamentBracket::query()
                ->whereIn('tournament_stage_id', $tournament->stages()->pluck('id'))
                ->whereNotNull('match_id')
                ->exists();
    }

    private function assignFinalPlacements(Tournament $tournament): void
    {
        // For single-elim: placement 1 = winner of round-N bracket; placement 2 = loser of round-N bracket; etc.
        // For double-elim: 1 = grand-final winner; 2 = grand-final loser; 3 = L-bracket-final loser; etc.
        // For round-robin / swiss: placement = rank from tournament_standings
        //
        // For v1: write placement based on standings.rank for round-robin/swiss; for single/double-elim,
        // walk brackets backwards (final → semi → etc.) and assign placements 1/2/3-4/5-8 etc.
        //
        // Simplest v1 implementation: read tournament_standings.rank (populated by StandingsCalculatorService)
        // and write tournament_participants.placement = standings.rank

        $tournament->participants()->update(['placement' => null]);  // reset
        foreach ($tournament->standings()->whereNotNull('rank')->orderBy('rank')->get() as $standing) {
            $standing->participant->update(['placement' => $standing->rank]);
        }
    }
}
```

```php
<?php
declare(strict_types=1);
namespace App\Observers;

use App\Models\MatchResult;
use App\Services\BracketAdvancementService;

class MatchResultObserver
{
    /**
     * Fires on every MatchResult save. The BracketAdvancementService is no-op for non-tournament matches.
     * Pattern 7 Option A — observer over inline service call (RESEARCH recommendation).
     */
    public function saved(MatchResult $result): void
    {
        // Only fire when winner_clan_id is set (draws don't advance)
        if ($result->winner_clan_id === null) {
            return;
        }

        // Only fire on relevant attribute changes — guards against pure timestamp updates
        if (! $result->wasChanged(['winner_clan_id', 'a_score', 'b_score', 'recorded_at']) && ! $result->wasRecentlyCreated) {
            return;
        }

        app(BracketAdvancementService::class)->advance($result);
    }
}
```
