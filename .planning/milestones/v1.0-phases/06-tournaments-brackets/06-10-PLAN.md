```php
// PublicTournamentData (privacy-filtered; SVG-ready)
#[TypeScript]
final class PublicTournamentData extends Data
{
    public function __construct(
        public readonly string $id,
        public readonly string $slug,
        /** @var array<string, string|null> $title */
        public readonly array $title,
        /** @var array<string, string|null>|null $description */
        public readonly ?array $description,
        public readonly string $format,
        public readonly string $status,
        public readonly ?\DateTimeInterface $starts_at,
        public readonly ?\DateTimeInterface $ends_at,
        public readonly ?int $max_participants,
        public readonly int $participant_count,
        /** @var list<BracketNodeData> $nodes */
        public readonly array $nodes,
        /** @var list<BracketEdgeData> $edges */
        public readonly array $edges,
        /** @var list<TournamentStandingData>|null $standings */
        public readonly ?array $standings,
        /** @var list<TournamentParticipantData>|null $participants */
        public readonly ?array $participants,
        public readonly string $etag,
        public readonly \DateTimeInterface $last_modified_at,
    ) {}

    public static function fromModel(Tournament $t): self
    {
        $nodes = [];
        $edges = [];
        foreach ($t->stages as $stage) {
            foreach ($stage->brackets as $bracket) {
                $nodes[] = BracketNodeData::fromModel($bracket);
                if ($bracket->advances_to_bracket_id !== null) {
                    $edges[] = new BracketEdgeData(
                        from_bracket_id: $bracket->id,
                        to_bracket_id: $bracket->advances_to_bracket_id,
                        to_slot: $bracket->position % 2 === 1 ? 'a' : 'b',
                        type: 'winner',
                    );
                }
                if ($bracket->loser_advances_to_bracket_id !== null) {
                    $edges[] = new BracketEdgeData(
                        from_bracket_id: $bracket->id,
                        to_bracket_id: $bracket->loser_advances_to_bracket_id,
                        to_slot: $bracket->position % 2 === 1 ? 'a' : 'b',
                        type: 'loser',
                    );
                }
            }
        }

        // ETag: sha1 of bracket-state serialised
        $etagSource = $t->updated_at?->toIso8601String() . '|' . $t->stages()->with('brackets')->get()->pluck('brackets')->flatten()->map(fn ($b) => $b->id . ':' . $b->updated_at?->toIso8601String())->implode(',');
        $etag = sha1($etagSource);

        return new self(
            id: $t->id,
            slug: $t->slug,
            title: $t->getTranslations('title'),
            description: $t->description !== null ? $t->getTranslations('description') : null,
            format: $t->format,
            status: $t->status,
            starts_at: $t->starts_at,
            ends_at: $t->ends_at,
            max_participants: $t->max_participants,
            participant_count: $t->participants()->where('status', '!=', 'registered')->count(),
            nodes: $nodes,
            edges: $edges,
            standings: $t->relationLoaded('standings')
                ? $t->standings->map(fn ($s) => TournamentStandingData::fromModel($s))->all()
                : null,
            participants: $t->relationLoaded('participants')
                ? $t->participants->map(fn ($p) => TournamentParticipantData::fromModel($p))->all()
                : null,
            etag: $etag,
            last_modified_at: $t->updated_at ?? now(),
        );
    }
}

// BracketNodeData (SVG render-shape per RESEARCH Pattern 8)
#[TypeScript]
final class BracketNodeData extends Data
{
    public function __construct(
        public readonly string $id,
        public readonly int $round_number,
        public readonly int $position,
        public readonly string $stage_type,  // 'elim'|'winners-bracket'|'losers-bracket'|'grand-final'|'group'|'swiss-round'
        public readonly ?ParticipantSummary $participant_a,
        public readonly ?ParticipantSummary $participant_b,
        public readonly ?string $winner_participant_id,
        public readonly ?string $match_id,
        public readonly string $status,  // 'pending'|'in-progress'|'completed'|'bye'
        public readonly ?\DateTimeInterface $scheduled_at,
    ) {}

    public static function fromModel(TournamentBracket $b): self
    {
        $status = $b->participant_a_id && !$b->participant_b_id && $b->winner_participant_id
            ? 'bye'
            : ($b->winner_participant_id !== null ? 'completed' : ($b->match_id !== null ? 'in-progress' : 'pending'));

        return new self(
            id: $b->id,
            round_number: $b->round_number,
            position: $b->position,
            stage_type: $b->stage->type,
            participant_a: $b->participantA ? new ParticipantSummary($b->participantA->id, $b->participantA->clan?->name ?? '', $b->participantA->seed ?? 0) : null,
            participant_b: $b->participantB ? new ParticipantSummary($b->participantB->id, $b->participantB->clan?->name ?? '', $b->participantB->seed ?? 0) : null,
            winner_participant_id: $b->winner_participant_id,
            match_id: $b->match_id,
            status: $status,
            scheduled_at: $b->match?->scheduled_start_at,
        );
    }
}
```

TournamentObserver real impl (replaces plan 06-03 stub):

```php
<?php
declare(strict_types=1);
namespace App\Observers;

use App\Models\DiscordOutboundMessage;
use App\Models\Event;
use App\Models\Tournament;
use App\Support\DiscordOutboundPayloadBuilder;

class TournamentObserver
{
    public function saved(Tournament $tournament): void
    {
        $shouldHaveEvent = $tournament->is_public && $tournament->status !== 'cancelled';
        if ($shouldHaveEvent) {
            Event::updateOrCreate(
                ['eventable_type' => Tournament::class, 'eventable_id' => $tournament->id],
                [
                    'starts_at' => $tournament->starts_at ?? now(),
                    'ends_at' => $tournament->ends_at,
                    'title' => $tournament->getTranslations('title'),
                    'is_public' => $tournament->is_public,
                ],
            );
            return;
        }
        Event::where('eventable_type', Tournament::class)
            ->where('eventable_id', $tournament->id)
            ->delete();
    }

    public function created(Tournament $tournament): void
    {
        if (!$tournament->is_public) return;
        $payload = app(DiscordOutboundPayloadBuilder::class)->buildTournamentAnnounce($tournament);
        DiscordOutboundMessage::create([
            'message_type' => 'tournament_announce',
            'status' => 'pending',
            'channel_id' => null,  // resolved at dispatch time
            'payload' => $payload,
            'causer_user_id' => auth()->id(),
        ]);
    }

    public function updated(Tournament $tournament): void
    {
        if (!$tournament->wasChanged('status')) return;
        if (!$tournament->is_public) return;
        $payload = app(DiscordOutboundPayloadBuilder::class)->buildTournamentAnnounce($tournament);
        DiscordOutboundMessage::create([
            'message_type' => 'tournament_announce_update',
            'status' => 'pending',
            'channel_id' => null,
            'payload' => $payload,
            'causer_user_id' => auth()->id(),
        ]);
    }
}
```
