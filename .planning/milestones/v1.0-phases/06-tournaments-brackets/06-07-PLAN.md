```php
final class RoundRobinGenerator implements BracketGeneratorStrategy
{
    public function generate(Tournament $tournament, Collection $orderedParticipants): void
    {
        $n = $orderedParticipants->count();
        if ($n < 2) {
            throw new \InvalidArgumentException(__('tournaments.errors.insufficient_participants', ['min' => 2]));
        }

        // Circle method: for odd N, append a ghost participant (null)
        $participants = $orderedParticipants->all();  // 0-indexed array
        $usedGhost = false;
        if ($n % 2 === 1) {
            $participants[] = null;  // ghost
            $usedGhost = true;
        }
        $nWithGhost = count($participants);
        $rounds = $nWithGhost - 1;
        $matchesPerRound = (int) ($nWithGhost / 2);

        // Stage 1: single group stage
        $stage = TournamentStage::create([
            'tournament_id' => $tournament->id,
            'type' => 'group',
            'ordinal' => 1,
            'name' => null,
            'settings' => null,
        ]);

        // Circle algorithm: fix participants[0], rotate participants[1..nWithGhost-1]
        // Round r (0-indexed): pair (fix, rotated[nWithGhost-2-r]); pair (rotated[(i-1+r) mod (nWithGhost-1)], rotated[(nWithGhost-2-i+r) mod (nWithGhost-1)]) for i in 1..matchesPerRound-1
        $fix = $participants[0];
        $rotating = array_slice($participants, 1);  // length = nWithGhost - 1

        for ($r = 0; $r < $rounds; $r++) {
            $position = 1;
            // First match: fix vs rotating[nWithGhost - 2 - r % (nWithGhost - 1)] — but standard circle is fix vs rotating[(0 + r) mod (nWithGhost-1)] ... use canonical circle algorithm:
            // Round r: pairs are (fix, rotating[r % (nWithGhost-1)]); then for i in 1..matchesPerRound-1: (rotating[(r + i) mod (nWithGhost-1)], rotating[(r - i + (nWithGhost-1)) mod (nWithGhost-1)])

            $pA = $fix;
            $pB = $rotating[$r % ($nWithGhost - 1)];
            if ($pA !== null && $pB !== null) {  // skip ghost-vs-fix pairings (impossible) and ghost-involved pairs
                TournamentBracket::create([
                    'tournament_stage_id' => $stage->id,
                    'round_number' => $r + 1,
                    'position' => $position++,
                    'participant_a_id' => $pA->id,
                    'participant_b_id' => $pB->id,
                    // No advances_to — round-robin has no advancement chain
                ]);
            }

            for ($i = 1; $i < $matchesPerRound; $i++) {
                $pA = $rotating[($r + $i) % ($nWithGhost - 1)];
                $pB = $rotating[($r - $i + ($nWithGhost - 1)) % ($nWithGhost - 1)];
                if ($pA !== null && $pB !== null) {
                    TournamentBracket::create([
                        'tournament_stage_id' => $stage->id,
                        'round_number' => $r + 1,
                        'position' => $position++,
                        'participant_a_id' => $pA->id,
                        'participant_b_id' => $pB->id,
                    ]);
                }
            }
        }
    }
}
```

```php
final class SwissGenerator implements BracketGeneratorStrategy
{
    /**
     * Generates round 1 ONLY. Round 2+ requires admin click via generateNextRound().
     * Open Question A6 RESOLVED LOCKED inline: admin-click for v1.
     */
    public function generate(Tournament $tournament, Collection $orderedParticipants): void
    {
        $n = $orderedParticipants->count();
        $rounds = (int) ceil(log($n, 2));

        // Pitfall 5: refuse to start swiss with too few participants for the required round count
        // 2^rounds is the lower bound — we need at least that many distinct participants
        $minRequired = 2 ** $rounds;
        if ($n < $minRequired) {
            throw new SwissTooFewParticipantsException(
                __('tournaments.errors.swiss_too_few_participants', ['min' => $minRequired, 'rounds' => $rounds])
            );
        }

        // Round 1: stage with type='swiss-round', ordinal=1
        $stage = TournamentStage::create([
            'tournament_id' => $tournament->id,
            'type' => 'swiss-round',
            'ordinal' => 1,
            'name' => __('tournaments.stage_types.swiss-round.label') . ' 1',
            'settings' => null,
        ]);

        // Round 1 pairing: top half vs bottom half by seed
        // For 8 participants: (seed 1 vs seed 5), (seed 2 vs seed 6), (seed 3 vs seed 7), (seed 4 vs seed 8)
        $half = (int) ($n / 2);
        $topHalf = $orderedParticipants->slice(0, $half)->values();
        $bottomHalf = $orderedParticipants->slice($half, $half)->values();

        for ($i = 0; $i < $half; $i++) {
            TournamentBracket::create([
                'tournament_stage_id' => $stage->id,
                'round_number' => 1,  // within-stage round (always 1 for a swiss-round stage)
                'position' => $i + 1,
                'participant_a_id' => $topHalf[$i]->id,
                'participant_b_id' => $bottomHalf[$i]->id,
            ]);
        }

        // Odd N: lowest-seed gets a bye (the leftover participant). RESEARCH Pattern 5.
        if ($n % 2 === 1) {
            $byeParticipant = $orderedParticipants->last();
            TournamentBracket::create([
                'tournament_stage_id' => $stage->id,
                'round_number' => 1,
                'position' => $half + 1,
                'participant_a_id' => $byeParticipant->id,
                'participant_b_id' => null,
                'winner_participant_id' => $byeParticipant->id,  // bye = auto-win
            ]);
        }
    }

    /**
     * Generates the next swiss round stage. Admin-click triggered (Open Question A6 RESOLVED LOCKED inline).
     * Pairs participants by (score DESC, buchholz DESC, seed ASC) within score groups; top half vs bottom half;
     * never-paired-before constraint with simple swap-down on conflict.
     */
    public function generateNextRound(Tournament $tournament): void
    {
        $currentRound = $tournament->stages()->where('type', 'swiss-round')->max('ordinal');
        $totalRounds = (int) ceil(log($tournament->participants()->where('status', 'active')->count(), 2));

        if ($currentRound >= $totalRounds) {
            throw new \LogicException(__('tournaments.errors.swiss_rounds_exhausted'));
        }

        $nextOrdinal = $currentRound + 1;

        // Fetch participants + their current scores + buchholz from tournament_standings
        $standings = $tournament->standings()
            ->orderBy('points', 'desc')
            ->orderBy('tiebreak_score', 'desc')
            ->get()
            ->each(function ($s) {
                $s->load('participant');
            });

        // Group by score (points)
        $grouped = $standings->groupBy('points');

        // Build pairing list: within each score group, pair top half vs bottom half; floats down on odd group
        $pairings = [];
        $alreadyPaired = $this->getAlreadyPairedSet($tournament);  // set of "smaller_id-larger_id" strings
        $floatDown = null;

        foreach ($grouped as $points => $group) {
            if ($floatDown !== null) {
                $group->prepend($floatDown);
                $floatDown = null;
            }
            if ($group->count() % 2 === 1) {
                $floatDown = $group->pop();  // bottom of this group floats down to next
            }
            $half = (int) ($group->count() / 2);
            $top = $group->slice(0, $half)->values();
            $bot = $group->slice($half, $half)->values();
            for ($i = 0; $i < $half; $i++) {
                $pA = $top[$i]->participant;
                $pB = $bot[$i]->participant;
                // Never-paired-before constraint — if duplicate, swap with next candidate
                $pairKey = $this->pairKey($pA->id, $pB->id);
                if (isset($alreadyPaired[$pairKey]) && $i + 1 < $half) {
                    // simple swap with next bottom; rerun
                    [$bot[$i], $bot[$i + 1]] = [$bot[$i + 1], $bot[$i]];
                    $pB = $bot[$i]->participant;
                    $pairKey = $this->pairKey($pA->id, $pB->id);
                }
                if (isset($alreadyPaired[$pairKey])) {
                    // Edge case — pair anyway with warning (admin can manually edit before round starts)
                    \Log::warning("Swiss pairing duplicate detected — tournament {$tournament->id} round {$nextOrdinal}: {$pA->id} vs {$pB->id}");
                }
                $pairings[] = [$pA, $pB];
                $alreadyPaired[$pairKey] = true;
            }
        }

        // If a participant floats all the way down with no pair, they get a bye
        $byeParticipant = null;
        if ($floatDown !== null) {
            $byeParticipant = $floatDown->participant;
        }

        // Create the next swiss-round stage + brackets
        $stage = TournamentStage::create([
            'tournament_id' => $tournament->id,
            'type' => 'swiss-round',
            'ordinal' => $nextOrdinal,
            'name' => __('tournaments.stage_types.swiss-round.label') . ' ' . $nextOrdinal,
            'settings' => null,
        ]);

        foreach ($pairings as $i => [$pA, $pB]) {
            TournamentBracket::create([
                'tournament_stage_id' => $stage->id,
                'round_number' => 1,
                'position' => $i + 1,
                'participant_a_id' => $pA->id,
                'participant_b_id' => $pB->id,
            ]);
        }

        if ($byeParticipant !== null) {
            TournamentBracket::create([
                'tournament_stage_id' => $stage->id,
                'round_number' => 1,
                'position' => count($pairings) + 1,
                'participant_a_id' => $byeParticipant->id,
                'participant_b_id' => null,
                'winner_participant_id' => $byeParticipant->id,
            ]);
        }
    }

    /** @return array<string, bool> */
    private function getAlreadyPairedSet(Tournament $tournament): array
    {
        $set = [];
        $brackets = $tournament->stages()
            ->where('type', 'swiss-round')
            ->with('brackets')
            ->get()
            ->pluck('brackets')
            ->flatten();
        foreach ($brackets as $b) {
            if ($b->participant_a_id && $b->participant_b_id) {
                $set[$this->pairKey($b->participant_a_id, $b->participant_b_id)] = true;
            }
        }
        return $set;
    }

    private function pairKey(string $a, string $b): string
    {
        return $a < $b ? "{$a}-{$b}" : "{$b}-{$a}";
    }
}
```
