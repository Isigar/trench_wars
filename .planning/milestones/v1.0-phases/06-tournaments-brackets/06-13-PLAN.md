```php
declare(strict_types=1);

it('every expected tournaments.php key resolves to a non-empty string', function () {

        // ... and so on for 16 format keys

        // ... 6 statuses × 2 keys
        // Participant status × 4
        // Errors × 8
        // Actions × 9 × ~4 = 36
        // Tabs × 5
        // Empty × 3
        // Stage types × 6
        // Nav × 1
    ];
    foreach ($expected as $key) {
        $resolved = __($key);
        expect($resolved)->not->toBe($key);  // key itself returned = i18n miss
        expect(trim((string) $resolved))->not->toBe('');
    }
});

it('every t() / __() call in apps/web/resources/js/pages/Tournaments + components/tournaments resolves to a defined key', function () {
    $sources = [
        ...glob(base_path('resources/js/pages/Tournaments/*.vue')),
        ...glob(base_path('resources/js/components/tournaments/*.vue')),
        ...glob(base_path('app/Filament/Resources/TournamentResource.php')),
        ...glob(base_path('app/Filament/Resources/TournamentResource/Pages/*.php')),
        ...glob(base_path('app/Filament/Resources/TournamentResource/RelationManagers/*.php')),
    ];
    $missing = [];
    foreach ($sources as $file) {
        $contents = file_get_contents($file);
        // Match t('tournaments.X') OR __('tournaments.X') OR __('admin.tournament.X')
        if (preg_match_all('/[t_]\\([\\s\\n]*[\'"]((?:tournaments|admin\\.tournament(?:_participant|_bracket|_standing)?)\\.[^\'"]+)[\'"]/', $contents, $m)) {
            foreach ($m[1] as $key) {
                $resolved = __($key);
                if (trim((string) $resolved) === '' || $resolved === $key) {
                    $missing[] = "$key (in $file)";
                }
            }
        }
    }
    expect($missing)->toBeEmpty('Missing i18n keys: ' . implode(', ', $missing));
});
```

Bot extension — apps/bot/src/services/embeds/tournament.ts (Phase 5 plan 05-10 embed idiom):

```typescript
import { EmbedBuilder } from 'discord.js'
import { Colors } from '../colors.js'

type TournamentAnnouncePayload = {
  tournament_id: string
  tournament_slug: string
  title: Record<string, string | null>
  format: string
  status: string
  starts_at: string | null
  ends_at: string | null
  organiser_user_id: string
  max_participants: number | null
  is_public: boolean
}

export function buildTournamentAnnounceEmbed(payload: TournamentAnnouncePayload, locale = 'en'): EmbedBuilder {
  const title = payload.title[locale] ?? payload.title.en ?? 'Tournament'
  const embed = new EmbedBuilder()
    .setTitle(title)
    .setColor(Colors.primary)
    .setURL(`${process.env.WEB_URL ?? 'http://localhost'}/tournaments/${payload.tournament_slug}`)
    .addFields(
      { name: 'Format', value: payload.format, inline: true },
      { name: 'Status', value: payload.status, inline: true },
      { name: 'Max participants', value: payload.max_participants?.toString() ?? '—', inline: true },
    )
  if (payload.starts_at !== null) {
    embed.addFields({ name: 'Starts at', value: `<t:${Math.floor(new Date(payload.starts_at).getTime() / 1000)}:F>` })
  }
  return embed
}

type BracketResultPayload = {
  tournament_id: string
  tournament_slug: string
  bracket_id: string
  round_number: number
  position: number
  winner_clan_id: string | null
  winner_clan_name: string | null
  participant_a_clan_name: string | null
  participant_b_clan_name: string | null
}

export function buildBracketResultEmbed(payload: BracketResultPayload, locale = 'en'): EmbedBuilder {
  const embed = new EmbedBuilder()
    .setTitle(`Round ${payload.round_number} — Match ${payload.position}`)
    .setColor(Colors.success)
    .setURL(`${process.env.WEB_URL ?? 'http://localhost'}/tournaments/${payload.tournament_slug}`)
    .setDescription(
      payload.winner_clan_name !== null
        ? `**${payload.winner_clan_name}** defeated ${payload.participant_a_clan_name === payload.winner_clan_name ? payload.participant_b_clan_name ?? '?' : payload.participant_a_clan_name ?? '?'}`
        : 'Result pending'
    )
  return embed
}
```
