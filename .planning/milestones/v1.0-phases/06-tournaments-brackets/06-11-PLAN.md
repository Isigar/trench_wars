```php
<?php
declare(strict_types=1);
namespace App\Filament\Resources\TournamentResource\Pages;

use App\Filament\Resources\TournamentResource;
use App\Models\Tournament;
use App\Services\Brackets\BracketGeneratorService;
use App\Services\Brackets\SwissGenerator;
use App\Services\BracketMatchMaterialiserService;
use App\Services\StandingsCalculatorService;
use App\Services\TournamentSeedingService;
use App\Services\TournamentStatusService;
use Filament\Actions;
use Filament\Forms;
use Filament\Notifications\Notification;
use Filament\Resources\Pages\EditRecord;

final class EditTournament extends EditRecord
{
    protected static string $resource = TournamentResource::class;

    protected function getHeaderActions(): array
    {
        /** @var Tournament $record */
        $record = $this->record;

        return [
            Actions\Action::make('open_registration')
                ->label(__('admin.tournament.actions.open_registration.label'))
                ->color('info')
                ->visible(fn () => $record->status === 'draft')
                ->requiresConfirmation()
                ->modalHeading(__('tournaments.actions.open_registration.modal_heading'))
                ->modalDescription(__('tournaments.actions.open_registration.modal_description'))
                ->action(function (): void {
                    app(TournamentStatusService::class)->transition($this->record, 'registering');
                    Notification::make()->success()->title(__('tournaments.actions.open_registration.success'))->send();
                }),

            Actions\Action::make('seed')
                ->label(__('admin.tournament.actions.seed.label'))
                ->color('primary')
                ->visible(fn () => $record->status === 'registering' && $record->participants()->where('status', 'registered')->count() >= 2)
                ->form([
                    Forms\Components\Select::make('strategy')
                        ->options([
                            'by_rank' => 'By rank',
                            'random' => 'Random',
                            'manual' => 'Manual',
                        ])
                        ->required(),
                ])
                ->requiresConfirmation()
                ->modalHeading(__('tournaments.actions.seed.modal_heading', ['count' => $record->participants()->where('status', 'registered')->count()]))
                ->action(function (array $data): void {
                    app(TournamentSeedingService::class)->seed($this->record, $data['strategy']);
                    app(TournamentStatusService::class)->transition($this->record->refresh(), 'seeded');
                    Notification::make()->success()->title(__('tournaments.actions.seed.success'))->send();
                }),

            Actions\Action::make('start')
                ->label(__('admin.tournament.actions.start.label'))
                ->color('success')
                ->visible(fn () => $record->status === 'seeded')
                ->requiresConfirmation()
                ->modalHeading(__('tournaments.actions.start.modal_heading'))
                ->modalDescription(__('tournaments.actions.start.modal_description'))
                ->action(function (): void {
                    app(BracketGeneratorService::class)->generate($this->record);
                    app(TournamentStatusService::class)->transition($this->record->refresh(), 'running');
                    app(BracketMatchMaterialiserService::class)->materialiseFirstRound($this->record->refresh());
                    Notification::make()->success()->title(__('tournaments.actions.start.success'))->send();
                }),

            Actions\Action::make('reseed')
                ->label(__('admin.tournament.actions.reseed.label'))
                ->color('warning')
                ->icon('heroicon-o-arrow-path')
                ->visible(fn () => $record->canReseed())
                ->form([
                    Forms\Components\Select::make('strategy')
                        ->options(['by_rank' => 'By rank', 'random' => 'Random', 'manual' => 'Manual'])
                        ->required(),
                ])
                ->requiresConfirmation()
                ->modalHeading(__('tournaments.actions.reseed.modal_heading'))
                ->modalDescription(__('tournaments.actions.reseed.modal_description'))
                ->action(function (array $data): void {
                    app(TournamentSeedingService::class)->reseed($this->record, $data['strategy']);
                    Notification::make()->success()->title(__('tournaments.actions.reseed.success'))->send();
                }),

            Actions\Action::make('generate_next_swiss_round')
                ->label(__('admin.tournament.actions.generate_next_swiss_round.label'))
                ->color('primary')
                ->visible(fn () => $record->status === 'running' && $record->format === 'swiss' && $this->isSwissRoundComplete())
                ->requiresConfirmation()
                ->action(function (): void {
                    app(SwissGenerator::class)->generateNextRound($this->record);
                    app(BracketMatchMaterialiserService::class)->materialiseFirstRound($this->record->refresh());  // materialise round n+1
                    Notification::make()->success()->title(__('tournaments.actions.generate_next_swiss_round.success'))->send();
                }),

            Actions\Action::make('materialise_next_round')
                ->label('Materialise next round')  // not in i18n — admin-only friction text; promote to lang/en/admin.php if surfaced as a real UX friction point
                ->color('info')
                ->visible(fn () => $record->status === 'running' && $record->format !== 'swiss' && $record->format !== 'round_robin' && $this->hasUnmaterialisedBrackets())
                ->requiresConfirmation()
                ->action(function (): void {
                    app(BracketMatchMaterialiserService::class)->materialiseFirstRound($this->record);  // method materialises EVERY round-1 bracket missing match_id; but for next-round we need a different method — see plan 06-06 BracketMatchMaterialiserService for the materialiseFor() per-bracket entry; here we call it for each unmaterialised bracket whose participants are both set
                    Notification::make()->success()->title('Round materialised.')->send();
                }),

            Actions\Action::make('recalculate_standings')
                ->label(__('admin.tournament.actions.recalculate_standings.label'))
                ->color('info')
                ->visible(fn () => in_array($record->status, ['running', 'completed'], true))
                ->requiresConfirmation()
                ->action(function (): void {
                    app(StandingsCalculatorService::class)->recalculate($this->record);
                    Notification::make()->success()->title(__('tournaments.actions.recalculate_standings.success'))->send();
                }),

            Actions\Action::make('cancel')
                ->label(__('admin.tournament.actions.cancel.label'))
                ->color('danger')
                ->visible(fn () => !in_array($record->status, ['completed', 'cancelled'], true))
                ->requiresConfirmation()
                ->modalHeading(__('tournaments.actions.cancel.modal_heading'))
                ->modalDescription(__('tournaments.actions.cancel.modal_description'))
                ->action(function (): void {
                    app(TournamentStatusService::class)->transition($this->record, 'cancelled');
                    Notification::make()->success()->title(__('tournaments.actions.cancel.success'))->send();
                }),
        ];
    }

    private function isSwissRoundComplete(): bool
    {
        $record = $this->record;
        $lastSwissStage = $record->stages()->where('type', 'swiss-round')->orderByDesc('ordinal')->first();
        if ($lastSwissStage === null) return false;
        return !$lastSwissStage->brackets()->whereNull('winner_participant_id')->exists();
    }

    private function hasUnmaterialisedBrackets(): bool
    {
        // ... brackets where match_id is null AND both participants are set
        return $this->record->stages()
            ->with('brackets')
            ->get()
            ->pluck('brackets')->flatten()
            ->where('match_id', null)
            ->whereNotNull('participant_a_id')
            ->whereNotNull('participant_b_id')
            ->isNotEmpty();
    }
}
```
