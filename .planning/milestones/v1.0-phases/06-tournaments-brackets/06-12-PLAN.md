```php
final class TournamentPublicJsonController extends Controller
{
    public function __invoke(\Illuminate\Http\Request $request, Tournament $tournament): \Illuminate\Http\JsonResponse
    {
        abort_unless($tournament->is_public, 404);
        $tournament->load([
            'stages.brackets.participantA.clan',
            'stages.brackets.participantB.clan',
            'standings.participant.clan',
            'participants.clan',
        ]);
        $data = \App\Data\PublicTournamentData::fromModel($tournament);

        $clientEtag = $request->header('If-None-Match');
        if ($clientEtag !== null && trim($clientEtag, '"') === $data->etag) {
            return response()->json(null, 304)->setEtag($data->etag);
        }

        return response()->json([
            'data' => $data,
            'etag' => $data->etag,
            'last_modified_at' => $data->last_modified_at,
        ])->setEtag($data->etag);
    }
}
```

BracketCanvas.vue (SVG renderer per Pattern 8 + Pitfall 9 stage-group offset):

```vue
<script setup lang="ts">
import { computed } from 'vue'
import type { BracketNodeData, BracketEdgeData } from '@trenchwars/shared-types'

const props = defineProps<{
  nodes: BracketNodeData[]
  edges: BracketEdgeData[]
}>()

const COLUMN_WIDTH = 220
const ROW_HEIGHT = 80
const NODE_WIDTH = 200
const NODE_HEIGHT = 60

// Group nodes by stage_type (Pitfall 9 — double-elim renders W + L + GF in separate groups)
const nodesByStage = computed(() => {
  const map = new Map<string, BracketNodeData[]>()
  for (const n of props.nodes) {
    if (!map.has(n.stage_type)) map.set(n.stage_type, [])
    map.get(n.stage_type)!.push(n)
  }
  return map
})

// Compute position map keyed by node.id
const positions = computed(() => {
  const map = new Map<string, { x: number, y: number }>()
  let stageYOffset = 0
  for (const [stageType, stageNodes] of nodesByStage.value) {
    const maxRound = Math.max(...stageNodes.map(n => n.round_number))
    for (const n of stageNodes) {
      const x = (n.round_number - 1) * COLUMN_WIDTH + 20
      const verticalSpacing = ROW_HEIGHT * Math.pow(2, n.round_number - 1)
      const y = stageYOffset + (n.position - 1) * verticalSpacing + verticalSpacing / 2
      map.set(n.id, { x, y })
    }
    // Increase offset for next stage group (Pitfall 9)
    const stageHeight = Math.pow(2, maxRound - 1) * ROW_HEIGHT * stageNodes.filter(n => n.round_number === 1).length
    stageYOffset += stageHeight + 100
  }
  return map
})

const viewBox = computed(() => {
  let maxX = 0, maxY = 0
  positions.value.forEach(p => { maxX = Math.max(maxX, p.x + NODE_WIDTH); maxY = Math.max(maxY, p.y + NODE_HEIGHT) })
  return `0 0 ${maxX + 20} ${maxY + 20}`
})
</script>

<template>
  <svg :viewBox="viewBox" class="w-full">
    <!-- Render edges first so they sit behind nodes -->
    <g class="edges">
      <line
        v-for="edge in edges"
        :key="`${edge.from_bracket_id}-${edge.to_bracket_id}-${edge.type}`"
        :x1="(positions.get(edge.from_bracket_id)?.x ?? 0) + 200"
        :y1="positions.get(edge.from_bracket_id)?.y ?? 0"
        :x2="positions.get(edge.to_bracket_id)?.x ?? 0"
        :y2="positions.get(edge.to_bracket_id)?.y ?? 0"
        :stroke="edge.type === 'loser' ? 'var(--color-bracket-loser-line)' : 'var(--color-bracket-winner-line)'"
        stroke-width="2"
      />
    </g>
    <g class="nodes">
      <g v-for="node in nodes" :key="node.id" :transform="`translate(${positions.get(node.id)?.x ?? 0},${positions.get(node.id)?.y ?? 0})`">
        <rect width="200" height="60" :fill="node.status === 'completed' ? 'var(--color-bracket-node-completed)' : 'var(--color-bracket-node-pending)'" />
        <text x="10" y="20" font-size="12">{{ node.participant_a?.clan_name ?? '—' }}</text>
        <text x="10" y="40" font-size="12">{{ node.participant_b?.clan_name ?? '—' }}</text>
      </g>
    </g>
  </svg>
</template>
```

useTournamentPolling.ts:

```typescript
import { ref, onMounted, onUnmounted, Ref } from 'vue'
import type { PublicTournamentData } from '@trenchwars/shared-types'

export function useTournamentPolling(slug: string, tournament: Ref<PublicTournamentData>): void {
  const lastEtag = ref<string | null>(tournament.value.etag)

  const poll = async () => {
    try {
      const res = await fetch(`/tournaments/${slug}.json`, {
        headers: lastEtag.value ? { 'If-None-Match': `"${lastEtag.value}"` } : {},
      })
      if (res.status === 304) return
      const body = await res.json()
      tournament.value = body.data
      lastEtag.value = body.etag
    } catch (e) {
      console.error('Tournament polling failed:', e)
    }
  }

  let intervalId: ReturnType<typeof setInterval> | null = null
  onMounted(() => {
    intervalId = setInterval(poll, 30000)
  })
  onUnmounted(() => {
    if (intervalId !== null) clearInterval(intervalId)
  })
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Public JSON endpoint to viewers | Untrusted Internet — rate-limited via throttle:60,1; ETag prevents amplification |
| PublicTournamentData privacy | Already privacy-filtered in plan 06-10 (no MVP PII in v1; participants surface clan names which are public per D-018) |
| Polling vs WebSocket | 30s polling is the explicit v1 choice (REQ-non-goals-round-1 CON-tournament-public-view); WebSocket upgrade is TOUR-V2-01 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-12-01 | Denial of Service | Hot polling endpoint DoS — many viewers refresh aggressively | mitigate | Laravel throttle:60,1 middleware (60 req/min/IP); ETag short-circuits 304 to bypass DB query cost on unchanged data; Phase 9 polish: HTTP CDN cache layer |
| T-06-12-02 | Tampering | Etag bypass — client sends a fake etag and forces 200 every poll | accept | Etag is advisory; without matching, server emits 200 + full payload (same as no-etag client); rate limit catches abusers |
| T-06-12-03 | Information Disclosure | Public viewer sees admin-only fields | mitigate | TournamentPublicJsonController only emits PublicTournamentData (privacy-filtered DTO, plan 06-10); no admin-only fields surface |
| T-06-12-04 | Tampering | Polling response sets cookies or sensitive headers | accept | Pure JSON response; Laravel default response doesn't set session cookies on JSON routes without explicit middleware; verify via Pest assertion |
| T-06-12-05 | Tampering | Vue polling continues after navigation away from page | mitigate | onUnmounted clears the setInterval; Vue unmount cleans up automatically |
| T-06-12-06 | Information Disclosure | Bracket SVG layout reveals internal IDs via DOM | accept | Bracket UUIDs are not sensitive (public route already exposes /tournaments/{slug} bracket IDs); IDs are not predictable enough to enable enumeration attacks |
</threat_model>
