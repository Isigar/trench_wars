<objective>
Wave 9: ship the bot-side worker + reconciliation layer — (a) the outbound polling worker that picks up pending discord_outbound_messages rows and delivers them to Discord, (b) the guildMemberUpdate event handler that reconciles Discord-side role changes back to the web (SC-4 second half).

The outbound worker is RESEARCH Pattern 4 verbatim. Every 5 seconds: GET /outbound-messages?status=pending&limit=20 (which atomically claims them via plan 05-04's lockForUpdate transaction); for each claimed row, render via plan 05-10's matchCard (for match_announce) or directly perform a REST role add/remove (for role_sync); then POST /outbound-messages/{id}/sent or /failed.

Critical safety properties:
1. Overlap-skip guard — if a tick is still running, drop the new one (prevents N concurrent fetches if one hangs)
2. Per-row try/catch — one failing row doesn't break the loop
3. Token scrub via plan 05-08's api.ts (Pitfall 3 — error messages never contain plaintext token)
4. allowed_mentions={parse:[]} on every Discord send (T-05-10-07 defence-in-depth)

The guildMemberUpdate handler (RESEARCH Pattern 6 verbatim) is the SC-4 reconciliation side. When a server admin manually adds/removes a clan role in Discord (bypassing the website), the bot diffs the roles cache and POSTs each delta to /api/bot/discord-events/role-change. Plan 05-04's BotApiDiscordEventController applies Pitfall 10 echo-suppression (60s window) to avoid infinite loops with the bot's own role-sync deliveries.

The ready event handler (plan 05-09) is amended to launch the worker on bot startup. The index.ts is amended to register the guildMemberUpdate listener.

Purpose: SC-3 (outbound delivery loop active) + SC-4 second half (Discord-side drift reconciliation).

apps/bot/src/services/outbound.ts (RESEARCH Pattern 4 verbatim with extensions):

  import type { Client } from 'discord.js';
  import { api } from './api';
  import { render } from './render';
  import { env } from '../env';
  import type { OutboundRow } from '../types/apiContracts';

  let running = false;

  export function startOutboundWorker(client: Client, intervalMs = env.OUTBOUND_POLL_INTERVAL_MS): NodeJS.Timeout {
      console.log(`[bot/outbound] Starting poll loop every ${intervalMs}ms`);
      return setInterval(async () => {
          if (running) return;
          running = true;
          try {
              const rows = await api.get<OutboundRow[]>('/outbound-messages?status=pending&limit=20');
              for (const row of rows) {
                  try {
                      const result = await render(client, row);
                      await api.post(`/outbound-messages/${row.id}/sent`, {
                          sent_message_id: result.discordMessageId,
                      });
                  } catch (err) {
                      const message = err instanceof Error ? err.message : String(err);
                      try {
                          await api.post(`/outbound-messages/${row.id}/failed`, {
                              last_error: message.slice(0, 2000),
                          });
                      } catch (ackErr) {
                          console.error('[bot/outbound] markFailed failed:', ackErr);
                      }
                  }
              }
          } catch (err) {
              console.error('[bot/outbound] poll tick error:', err);
          } finally {
              running = false;
          }
      }, intervalMs);
  }

apps/bot/src/services/render.ts:

  import type { Client, TextBasedChannel } from 'discord.js';
  import { matchCard } from '../lib/embeds';
  import { env } from '../env';
  import type { OutboundRow, PublicMatchData } from '../types/apiContracts';

  interface RenderResult {
      discordMessageId: string;
  }

  export async function render(client: Client, row: OutboundRow): Promise<RenderResult> {
      if (row.message_type === 'match_announce') {
          return renderMatchAnnounce(client, row);
      }
      if (row.message_type === 'role_sync') {
          return renderRoleSync(client, row);
      }
      throw new Error(`[bot/render] Unknown message_type: ${row.message_type}`);
  }

  async function renderMatchAnnounce(client: Client, row: OutboundRow): Promise<RenderResult> {
      // payload shape from plan 05-05 DiscordOutboundPayloadBuilder
      const payload = row.payload as Record<string, unknown>;
      const matchId = payload.match_id as string;
      // Re-fetch the match to get full DTO with slot data
      const match = await api.get<PublicMatchData>(`/matches/${matchId}`);

      const channel = await client.channels.fetch(row.channel_id);
      if (channel === null || !channel.isTextBased() || !('send' in channel)) {
          throw new Error(`[bot/render] Channel ${row.channel_id} not found or not text-based`);
      }

      const { embeds, components } = matchCard(match);

      if (payload.kind === 'match_announce_update' && payload.prior_sent_message_id !== null) {
          // Edit the existing message
          const priorId = payload.prior_sent_message_id as string;
          try {
              const msg = await (channel as any).messages.fetch(priorId);
              await msg.edit({ embeds, components, allowed_mentions: { parse: [] } });
              return { discordMessageId: priorId };
          } catch {
              // Fallthrough: prior message gone, post a new one
          }
      }

      const sent = await (channel as any).send({ embeds, components, allowed_mentions: { parse: [] } });
      return { discordMessageId: sent.id };
  }

  async function renderRoleSync(client: Client, row: OutboundRow): Promise<RenderResult> {
      const payload = row.payload as Record<string, unknown>;
      const userId = payload.user_discord_id as string;
      const roleId = payload.role_discord_id as string;
      const action = payload.action as 'add' | 'remove';

      const guild = await client.guilds.fetch(env.DISCORD_GUILD_ID);
      const member = await guild.members.fetch(userId).catch(() => null);
      if (member === null) {
          throw new Error(`[bot/render] Guild member ${userId} not found in guild ${env.DISCORD_GUILD_ID}`);
      }

      if (action === 'add') {
          await member.roles.add(roleId);
      } else {
          await member.roles.remove(roleId);
      }

      // Role sync has no Discord message id; return synthetic value so markSent ack succeeds
      return { discordMessageId: `role_sync:${action}:${userId}:${roleId}` };
  }

apps/bot/src/events/guildMemberUpdate.ts (RESEARCH Pattern 6 verbatim):

  import { Events, GuildMember, type PartialGuildMember, Client } from 'discord.js';
  import { api } from '../services/api';

  export function registerGuildMemberUpdateHandler(client: Client): void {
      client.on(Events.GuildMemberUpdate, async (oldMember, newMember) => {
          try {
              const removed = oldMember.roles.cache.filter((r) => !newMember.roles.cache.has(r.id));
              const added = newMember.roles.cache.filter((r) => !oldMember.roles.cache.has(r.id));

              for (const [, role] of removed) {
                  await api.post('/discord-events/role-change', {
                      user_discord_id: newMember.id,
                      role_discord_id: role.id,
                      action: 'remove',
                  });
              }
              for (const [, role] of added) {
                  await api.post('/discord-events/role-change', {
                      user_discord_id: newMember.id,
                      role_discord_id: role.id,
                      action: 'add',
                  });
              }
          } catch (err) {
              console.error('[bot/guildMemberUpdate]', err);
          }
      });
  }
