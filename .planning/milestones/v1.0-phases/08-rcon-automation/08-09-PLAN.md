CrconHealthProbe (stateless):
```php
final class CrconHealthProbe
{
    /** @return array{status: 'ok'|'error', error: ?string, map_rotation: ?array} */
    public function probe(MatchServer $server): array
    {
        try {
            $token = $server->credentials_encrypted['api_token'] ?? null;
            if (!$token) {
                return ['status' => 'error', 'error' => __('rcon.errors.permission_denied'), 'map_rotation' => null];
            }
            $url = sprintf('http://%s:%d/api/get_map_rotation', $server->host, $server->port_rcon);
            $resp = Http::withToken($token)->timeout(10)->get($url);
            if ($resp->failed()) {
                $kind = $resp->status() === 401 ? 'auth_failed' : 'unreachable';
                return ['status' => 'error', 'error' => __("rcon.errors.{$kind}"), 'map_rotation' => null];
            }
            return ['status' => 'ok', 'error' => null, 'map_rotation' => $resp->json('result')];
        } catch (\Throwable $e) {
            // ConnectionException, etc.
            return ['status' => 'error', 'error' => __('rcon.errors.unreachable'), 'map_rotation' => null];
        }
    }
}
```

```php
final class TestMatchServerConnectionJob implements ShouldQueue
{
    use Queueable;
    public function __construct(public string $matchServerId) {}
    public function handle(CrconHealthProbe $probe): void {
        $server = MatchServer::findOrFail($this->matchServerId);
        $result = $probe->probe($server);
        $server->update([
            'last_test_at' => now(),
            'last_test_status' => $result['status'],
            'last_test_error' => $result['error'],
        ]);
    }
}
```
