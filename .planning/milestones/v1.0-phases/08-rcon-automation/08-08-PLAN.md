```php
public function upsertFromRcon(GameMatch $match, array $data): MatchResult
{
    return DB::transaction(function () use ($match, $data): MatchResult {
        $existing = MatchResult::where('match_id', $match->id)->first();

        // INVARIANT: Manual override always wins.
        if ($existing && $existing->source === 'manual') {
            activity()
                ->performedOn($existing)
                ->withProperties([
                    'event' => 'rcon.arrived_but_manual_locked',
                    'would_have_set' => [
                        'winner_clan_id' => $data['winner_clan_id'] ?? null,
                        'allies_score' => $data['allies_score'] ?? null,
                        'axis_score' => $data['axis_score'] ?? null,
                    ],
                ])
                ->log(__('rcon.audit.rcon_arrived_locked'));
            return $existing;
        }

        $rconUser = User::where('email', 'rcon-worker@system.trenchwars')->firstOrFail();
        $result = MatchResult::updateOrCreate(
            ['match_id' => $match->id],
            [
                'winner_clan_id' => $data['winner_clan_id'] ?? null,
                'allies_score' => $data['allies_score'] ?? null,
                'axis_score' => $data['axis_score'] ?? null,
                'notes' => $data['notes'] ?? __('rcon.audit.automated_from_crcon'),
                'recorded_by_user_id' => $rconUser->id,
                'recorded_at' => $data['recorded_at'] ?? now(),
                'source' => 'rcon',
            ],
        );

        if ($match->status !== 'played') {
            app(MatchStatusService::class)->transition($match, 'played', $rconUser);
        }
        return $result;
    });
}
```

MatchPlayerStatAggregator:
```php
final class MatchPlayerStatAggregator
{
    public function aggregate(GameMatch $match): int // returns rows-upserted
    {
        return DB::transaction(function () use ($match): int {
            $events = MatchEvent::where('match_id', $match->id)
                ->whereIn('event_type', ['player_kill','player_team_kill','player_connect','player_disconnect','team_switch'])
                ->get();

            /** @var array<string, array<string, int|array<string,int>>> $perPlayer */
            $perPlayer = [];

            foreach ($events as $e) {
                $payload = $e->payload;
                switch ($e->event_type) {
                    case 'player_kill':
                        $kSteam = $payload['killer']['steam_id_64'] ?? null;
                        $vSteam = $payload['victim']['steam_id_64'] ?? null;
                        if ($kSteam) {
                            $perPlayer[$kSteam]['kills'] = ($perPlayer[$kSteam]['kills'] ?? 0) + 1;
                            $weapon = $payload['weapon'] ?? 'unknown';
                            $perPlayer[$kSteam]['weapons_used'][$weapon] = ($perPlayer[$kSteam]['weapons_used'][$weapon] ?? 0) + 1;
                        }
                        if ($vSteam) {
                            $perPlayer[$vSteam]['deaths'] = ($perPlayer[$vSteam]['deaths'] ?? 0) + 1;
                        }
                        break;
                    case 'player_team_kill':
                        $kSteam = $payload['killer']['steam_id_64'] ?? null;
                        if ($kSteam) {
                            $perPlayer[$kSteam]['team_kills'] = ($perPlayer[$kSteam]['team_kills'] ?? 0) + 1;
                        }
                        break;
                    // player_connect/disconnect/team_switch — could record role_played later; for round-1 just ensure player row exists
                }
            }

            $upserted = 0;
            foreach ($perPlayer as $steamId => $stats) {
                $player = Player::firstWhere('steam_id_64', $steamId);
                if (!$player) continue; // Pitfall 5 — orphan event silently skipped
                MatchPlayerStat::updateOrCreate(
                    ['match_id' => $match->id, 'player_id' => $player->id],
                    [
                        'kills' => $stats['kills'] ?? 0,
                        'deaths' => $stats['deaths'] ?? 0,
                        'team_kills' => $stats['team_kills'] ?? 0,
                        'score' => ($stats['kills'] ?? 0) * 100, // CRCON does not emit per-player score in /ws/logs
                        'weapons_used' => $stats['weapons_used'] ?? null,
                    ],
                );
                $upserted++;
            }
            return $upserted;
        });
    }
}
```

CloseMatchJob::handle (fills the placeholder from plan 08-07):
```php
public function handle(MatchPlayerStatAggregator $aggregator, MatchResultService $resultService): void
{
    $match = GameMatch::findOrFail($this->matchId);

    // 1. Aggregate stats
    $aggregator->aggregate($match);

    // 2. Derive result from match_end event
    $endEvent = MatchEvent::where('match_id', $match->id)
        ->where('event_type', 'match_end')
        ->latest('occurred_at')
        ->first();

    if (!$endEvent) {
        // Failure path: no match_end captured — flag for manual entry (RESEARCH failure table)
        $match->update(['manual_entry_required' => true]);
        return;
    }

    $resultData = [
        'allies_score' => $endEvent->payload['allies_score'] ?? null,
        'axis_score' => $endEvent->payload['axis_score'] ?? null,
        'winner_clan_id' => null, // round-1: cannot map team → clan deterministically; admin curates after
        'recorded_at' => $endEvent->occurred_at,
    ];

    // Low-confidence flag: if there were zero kills, manual entry required (RESEARCH "Match end with no MATCH ENDED event captured" row)
    $killCount = MatchEvent::where('match_id', $match->id)
        ->whereIn('event_type', ['player_kill'])
        ->count();
    if ($killCount === 0) {
        $match->update(['manual_entry_required' => true]);
    }

    $resultService->upsertFromRcon($match, $resultData);
}
```
