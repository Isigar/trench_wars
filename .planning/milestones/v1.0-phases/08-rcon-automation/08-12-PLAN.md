```php
use App\Models\{Clan, GameMatch, MatchServer, MatchServerBooking, Player, User};
use App\Models\MatchEvent;
use App\Models\MatchPlayerStat;
use App\Models\MatchResult;
use App\Models\DiscordOutboundMessage;
use Tests\Support\SignsRconRequests;

uses(SignsRconRequests::class);

test('SC-5: two clans complete the full round-1 happy path end-to-end', function () {
    // 1. Seed: two clans (clanA, clanB), two players each (with Steam IDs populated — Pitfall 5)
    $clanA = Clan::factory()->create(['name' => 'Wolves']);
    $clanB = Clan::factory()->create(['name' => 'Hawks']);
    [$alice, $aliceUser] = $this->playerWithSteamId('111', $clanA);
    [$bob, $bobUser]     = $this->playerWithSteamId('222', $clanA);
    [$carl, $carlUser]   = $this->playerWithSteamId('333', $clanB);
    [$dave, $daveUser]   = $this->playerWithSteamId('444', $clanB);

    // 2. Schedule scrim (Phase 4 factories)
    $match = GameMatch::factory()
        ->forHostClan($clanA)
        ->forMatchType($scrim)
        ->state(['status' => 'open_for_signups', 'scheduled_start' => now()->addHour(), 'scheduled_end' => now()->addHours(3), 'host_clan_id' => $clanA->id])
        ->create();
    // ... materialise slots (MatchSlotMaterialiser) — already happens on create()

    // 3. Signups via service (skip bot HTTP layer for test brevity — service-layer is the source of truth)
    app(MatchSignupService::class)->signup($match, $alice, $rifleSlot);
    // ... × 4 players to 4 slots ...

    // 4. Register server + book
    $server = MatchServer::factory()->create();
    MatchServerBooking::factory()->forMatch($match)->onServer($server)->create([
        'reserved_from' => $match->scheduled_start->subMinutes(5),
        'reserved_to'   => $match->scheduled_end->addMinutes(30),
    ]);

    // 5. Simulate CRCON events POSTed via HMAC-signed request
    $events = [
        MatchEventFactory::wireShape()->gameStart()->make(),
        MatchEventFactory::wireShape()->kill('111','333','K98')->make(),  // Alice → Carl
        MatchEventFactory::wireShape()->kill('111','444','K98')->make(),  // Alice → Dave
        MatchEventFactory::wireShape()->kill('222','333','M1A1')->make(), // Bob → Carl
        MatchEventFactory::wireShape()->kill('333','222','MP40')->make(), // Carl → Bob (axis comeback)
        MatchEventFactory::wireShape()->matchEnd('allies', 3, 1)->make(),
    ];
    $this->signedJsonPost(route('internal.match.events.store', $match), ['events' => $events])
        ->assertStatus(202);

    // 6. Drive Horizon to process the dispatched CloseMatchJob
    Queue::assertPushed(CloseMatchJob::class);
    Bus::dispatchSync(new CloseMatchJob($match->id)); // synchronous for test determinism

    // 7. Assertions
    $match->refresh();
    expect($match->status)->toBe('played');
    expect($match->manual_entry_required)->toBeFalse();

    $result = MatchResult::where('match_id', $match->id)->first();
    expect($result)->not->toBeNull();
    expect($result->source)->toBe('rcon');
    expect($result->allies_score)->toBe(3);
    expect($result->axis_score)->toBe(1);

    expect(MatchPlayerStat::where('match_id', $match->id)->count())->toBe(4); // 4 players in stats
    $aliceStats = MatchPlayerStat::where(['match_id'=>$match->id,'player_id'=>$alice->id])->first();
    expect($aliceStats->kills)->toBe(2);
    expect($aliceStats->deaths)->toBe(0);

    // 8. Discord result announce
    expect(DiscordOutboundMessage::where('match_id', $match->id)->where('message_type','match_result_announce')->count())->toBe(1);
});
```

DiscordOutboundPayloadBuilder::buildMatchResultAnnounce:
```php
public function buildMatchResultAnnounce(MatchResult $result): array
{
    $stats = MatchPlayerStat::with('player')
        ->where('match_id', $result->match_id)
        ->orderByRaw('(kills - deaths) DESC')
        ->take(3)
        ->get();

    return [
        'embeds' => [[
            'title' => __('rcon.embed.match_result.title', ['match' => $result->match_id]),
            'color' => 0x9b2c3d,
            'fields' => [
                ['name' => __('rcon.embed.match_result.score'), 'value' => "{$result->allies_score} - {$result->axis_score}", 'inline' => true],
                ['name' => __('rcon.embed.match_result.winner'), 'value' => $result->winnerClan?->name ?? '—', 'inline' => true],
                ['name' => __('rcon.embed.match_result.mvps'),
                 'value' => $stats->map(fn ($s) => "{$s->player->username}: K{$s->kills}/D{$s->deaths}")->implode("\n"),
                 'inline' => false],
            ],
        ]],
    ];
}
```
