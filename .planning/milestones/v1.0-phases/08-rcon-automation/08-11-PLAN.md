BookingScheduler:
```typescript
export class BookingScheduler {
  private active = new Map<string, MatchLifecycleManager>(); // bookingId -> manager
  private timer?: NodeJS.Timeout;

  constructor(private opts: {
    webClient: WebIngestClient;
    logger: Logger;
    redis: Redis; // ioredis
    secret: string;
    pollIntervalMs: number;
  }) {}

  start(): void {
    this.timer = setInterval(() => this.tick(), this.opts.pollIntervalMs);
    this.tick(); // immediate first poll
  }

  stop(): void {
    clearInterval(this.timer);
    for (const m of this.active.values()) m.stop();
    this.active.clear();
  }

  private async tick(): Promise<void> {
    try {
      const due = await this.opts.webClient.fetchSignedJson('/api/internal/bookings/due');
      // due: BookingDueData[]
      for (const b of due) {
        if (!this.active.has(b.id)) {
          const mgr = new MatchLifecycleManager({ booking: b, webClient: this.opts.webClient, logger: this.opts.logger, redis: this.opts.redis, secret: this.opts.secret });
          this.active.set(b.id, mgr);
          await mgr.start();
        }
      }
      // Reap completed bookings
      for (const [bookingId, mgr] of this.active.entries()) {
        if (mgr.isComplete()) {
          this.active.delete(bookingId);
        }
      }
    } catch (e) {
      this.opts.logger.warn({ err: (e as Error).message }, 'booking poll failed');
    }
  }
}
```

```typescript
export class MatchLifecycleManager {
  private crcon?: CrconClient;
  private buffer: NormalisedEvent[] = [];
  private flushTimer?: NodeJS.Timeout;
  private complete = false;
  private sawMatchEnd = false;
  private reservedToMs: number;

  constructor(private opts: {
    booking: BookingDueData;
    webClient: WebIngestClient;
    logger: Logger;
    redis: Redis;
    secret: string;
  }) {
    this.reservedToMs = new Date(opts.booking.reserved_to).getTime();
  }

  async start(): Promise<void> {
    let creds: { host: string; port_rcon: number; api_token: string };
    try {
      creds = await this.opts.webClient.fetchSignedJson(`/api/internal/match-servers/${this.opts.booking.server_id}/credentials`);
    } catch (e) {
      await this.emitManualError('unreachable', (e as Error).message);
      this.complete = true;
      return;
    }
    this.crcon = new CrconClient({
      url: `ws://${creds.host}:${creds.port_rcon}/ws/logs`,
      token: creds.api_token,
      logger: this.opts.logger,
      onLogs: (logs, _lastSeenId) => this.onLogs(logs),
      onError: (err) => this.opts.logger.warn({ err: err.message }, 'crcon error'),
    });
    this.crcon.connect();
    this.flushTimer = setInterval(() => this.flush(), 2_000);
    setTimeout(() => this.tryComplete(), Math.max(0, this.reservedToMs + 60_000 - Date.now()));
  }

  stop(): void {
    clearInterval(this.flushTimer);
    this.crcon?.close();
  }

  isComplete(): boolean { return this.complete; }

  private onLogs(logs: unknown[]): void {
    for (const raw of logs) {
      const evt = normalise(raw as { id: string; log: Record<string, unknown> });
      if (!evt) continue;
      this.buffer.push(evt);
      if (evt.event_type === 'match_end') this.sawMatchEnd = true;
    }
    if (this.buffer.length >= 10) void this.flush();
  }

  private async flush(): Promise<void> {
    if (this.buffer.length === 0) return;
    const batch = this.buffer.splice(0, this.buffer.length);
    const result = await this.opts.webClient.postEvents(this.opts.booking.match_id, batch).catch((e) => ({ status: 0, body: { err: (e as Error).message } }));
    if (result.status >= 200 && result.status < 300) return;
    // Failure: queue for retry
    await this.opts.redis.lpush(`rcon:queue:${this.opts.booking.match_id}`, JSON.stringify(batch));
    this.opts.logger.warn({ status: result.status, matchId: this.opts.booking.match_id, batchSize: batch.length }, 'web ingest failed; queued for retry');
  }

  private async emitManualError(kind: 'unreachable' | 'auth_failed' | 'permission_denied', detail: string): Promise<void> {
    const evt: NormalisedEvent = {
      event_type: 'manual_error',
      crcon_action: 'SYNTHETIC',
      crcon_stream_id: `manual_error-${Date.now()}-${this.opts.booking.id}`,
      occurred_at: new Date().toISOString(),
      payload: { kind, detail },
    };
    await this.opts.webClient.postEvents(this.opts.booking.match_id, [evt]).catch((e) => {
      this.opts.logger.error({ err: (e as Error).message }, 'failed to emit manual_error');
    });
  }

  private tryComplete(): void {
    if (!this.sawMatchEnd) {
      this.opts.logger.info({ matchId: this.opts.booking.match_id }, 'booking elapsed without match_end');
    }
    this.complete = true;
    this.stop();
  }
}
```

RedisFailoverQueue (drainer):
```typescript
export class RedisFailoverQueue {
  private timer?: NodeJS.Timeout;
  constructor(private opts: { redis: Redis; webClient: WebIngestClient; logger: Logger }) {}

  start(intervalMs = 5_000): void {
    this.timer = setInterval(() => this.drain(), intervalMs);
  }

  stop(): void { clearInterval(this.timer); }

  private async drain(): Promise<void> {
    // SCAN keys 'rcon:queue:*'
    const stream = this.opts.redis.scanStream({ match: 'rcon:queue:*', count: 50 });
    for await (const batch of stream) {
      for (const key of batch as string[]) {
        const matchId = key.replace('rcon:queue:', '');
        const items = await this.opts.redis.lrange(key, 0, 99);
        if (items.length === 0) continue;
        const events = items.flatMap((s) => { try { return JSON.parse(s); } catch { return []; } });
        const result = await this.opts.webClient.postEvents(matchId, events).catch(() => ({ status: 0, body: null }));
        if (result.status >= 200 && result.status < 300) {
          await this.opts.redis.ltrim(key, items.length, -1); // remove drained items
        } else {
          this.opts.logger.warn({ matchId, status: result.status }, 'failover drain failed');
        }
      }
    }
  }
}
```

```typescript
import { loadConfig } from './config.js';
import { logger } from './logging/logger.js';
import Redis from 'ioredis';
import { WebIngestClient } from './ingest/WebIngestClient.js';
import { BookingScheduler } from './booking/BookingScheduler.js';
import { RedisFailoverQueue } from './queue/RedisFailoverQueue.js';

async function main() {
  const cfg = loadConfig();
  const redis = new Redis(cfg.REDIS_URL || 'redis://redis:6379');
  const webClient = new WebIngestClient({ secret: cfg.WEB_HMAC_SECRET, baseUrl: cfg.WEB_INTERNAL_URL, logger });
  const scheduler = new BookingScheduler({ webClient, logger, redis, secret: cfg.WEB_HMAC_SECRET, pollIntervalMs: cfg.POLL_INTERVAL_MS });
  const drainer = new RedisFailoverQueue({ redis, webClient, logger });
  scheduler.start();
  drainer.start();
  logger.info('rcon-worker started');
  const shutdown = () => { scheduler.stop(); drainer.stop(); redis.disconnect(); process.exit(0); };
  process.on('SIGTERM', shutdown);
  process.on('SIGINT', shutdown);
}
main().catch((e) => { logger.fatal({ err: e.message }, 'fatal startup error'); process.exit(1); });
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| worker scheduler ↔ web internal API | All worker→web requests HMAC-signed. |
| worker → CRCON | Per-booking ephemeral session; token from web's encrypted store. |
| worker queue | Redis is internal Railway network only; no public surface. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-11-01 | Tampering | RedisFailoverQueue items mutated by attacker with Redis access | accept | Redis is internal to Railway; no public network exposure. Round-1 trust model. |
| T-08-11-02 | DoS | A single failing match queues unbounded events | mitigate | Drainer trims after success; LRANGE caps at 100 per drain pass so memory ceiling bounded; if queue grows forever it indicates a stuck web service — alerting via plan 08-12 audit. |
| T-08-11-03 | Spoofing | Worker scaled out replicas claim same booking | mitigate | Documented "do NOT scale rcon-worker beyond 1 replica in round 1" (Pitfall 11 / D-014); scheduler's per-process `this.active` Map prevents same-replica race. |
| T-08-11-04 | Information disclosure | manual_error payload contains exception stack | mitigate | emitManualError only sends `kind` + a short `detail` string (no stack); the worker logs the stack locally (pino) but does not transmit. |
</threat_model>
