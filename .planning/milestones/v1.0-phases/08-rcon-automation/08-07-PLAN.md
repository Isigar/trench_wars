MatchEventIngestService (final, container-resolvable, non-Mockery testable):
```php
final class MatchEventIngestService
{
    public function __construct(
        private MatchEventNormaliser $normaliser,
    ) {}

    /** @param array<int, array<string, mixed>> $events */
    public function ingest(GameMatch $match, array $events): array
    {
        $batchId = (string) Str::uuid();
        $accepted = 0;
        $skipped = 0;
        $sawMatchEnd = false;

        foreach ($events as $raw) {
            $dto = $this->normaliser->validate($raw); // throws on bad shape
            try {
                MatchEvent::create([
                    'match_id' => $match->id,
                    'event_type' => $dto->event_type,
                    'crcon_action' => $dto->crcon_action,
                    'crcon_stream_id' => $dto->crcon_stream_id,
                    'payload' => $dto->payload,
                    'occurred_at' => $dto->occurred_at,
                ]);
                $accepted++;
                if ($dto->event_type === 'match_end') {
                    $sawMatchEnd = true;
                }
            } catch (UniqueConstraintViolationException $e) {
                $skipped++; // idempotent — UNIQUE (match_id, crcon_stream_id) collision
            }
        }

        if ($sawMatchEnd) {
            Bus::dispatch(new CloseMatchJob($match->id));
        }

        return [
            'batch_id' => $batchId,
            'accepted_count' => $accepted,
            'skipped_count' => $skipped,
        ];
    }
}
```

```php
final class MatchEventNormaliser
{
    public function validate(array $event): MatchEventInputData {
        // Already validated by FormRequest at controller boundary; this re-validates payload shape per event_type:
        $type = $event['event_type'] ?? null;
        $payload = $event['payload'] ?? [];

        match ($type) {
            'player_kill', 'player_team_kill' => $this->assertKillShape($payload),
            'player_connect', 'player_disconnect' => $this->assertPlayerShape($payload),
            'team_switch' => $this->assertTeamSwitchShape($payload),
            'match_end', 'round_end' => $this->assertResultShape($payload),
            'game_start', 'round_start', 'manual_error' => null, // permissive
            default => throw new \InvalidArgumentException("unknown event_type: {$type}"),
        };

        return MatchEventInputData::from($event);
    }

    private function assertKillShape(array $p): void {
        if (!isset($p['killer']['steam_id_64'], $p['victim']['steam_id_64'], $p['weapon'])) {
            throw new \InvalidArgumentException('kill payload missing killer/victim/weapon');
        }
    }
    // ... similar private guards per RESEARCH canonical event-shape table
}
```
